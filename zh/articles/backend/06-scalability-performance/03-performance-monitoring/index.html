Performance monitoring and bottleneck analysis are essential practices for maintaining responsive, reliable backend systems. Proactive monitoring helps identify issues before they impact users.

Introduction

Performance bottlenecks can occur at any layer of your application stack—from slow database queries to memory leaks, network latency, or inefficient algorithms. Effective monitoring provides visibility into system behavior and enables data-driven optimization decisions.

Custom Metrics with Prometheus

from prometheus_client import (
    Counter, Histogram, Gauge, Summary,
    start_http_server, REGISTRY
)
from functools import wraps
import time
import psutil

# Define metrics
REQUEST_COUNT = Counter(
    'http_requests_total',
    'Total HTTP requests',
    ['method', 'endpoint', 'status']
)

REQUEST_LATENCY = Histogram(
    'http_request_duration_seconds',
    'HTTP request latency',
    ['method', 'endpoint'],
    buckets=[0.01, 0.05, 0.1, 0.25, 0.5, 1.0, 2.5, 5.0, 10.0]
)

ACTIVE_REQUESTS = Gauge(
    'http_requests_active',
    'Number of active HTTP requests'
)

DB_QUERY_LATENCY = Histogram(
    'db_query_duration_seconds',
    'Database query latency',
    ['query_type', 'table'],
    buckets=[0.001, 0.005, 0.01, 0.05, 0.1, 0.5, 1.0]
)

CACHE_HIT_RATE = Gauge(
    'cache_hit_rate',
    'Cache hit rate percentage'
)


def track_request_metrics(method: str, endpoint: str):
    """Decorator to track HTTP request metrics."""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            ACTIVE_REQUESTS.inc()
            start_time = time.perf_counter()
            status = "success"
            
            try:
                result = await func(*args, **kwargs)
                return result
            except Exception as e:
                status = "error"
                raise
            finally:
                duration = time.perf_counter() - start_time
                ACTIVE_REQUESTS.dec()
                REQUEST_COUNT.labels(
                    method=method,
                    endpoint=endpoint,
                    status=status
                ).inc()
                REQUEST_LATENCY.labels(
                    method=method,
                    endpoint=endpoint
                ).observe(duration)
        return wrapper
    return decorator


class DatabaseMetrics:
    """Track database query performance."""
    
    @staticmethod
    def track_query(query_type: str, table: str):
        def decorator(func):
            @wraps(func)
            async def wrapper(*args, **kwargs):
                start_time = time.perf_counter()
                try:
                    return await func(*args, **kwargs)
                finally:
                    duration = time.perf_counter() - start_time
                    DB_QUERY_LATENCY.labels(
                        query_type=query_type,
                        table=table
                    ).observe(duration)
            return wrapper
        return decorator


class SystemMetricsCollector:
    """Collect system-level metrics."""
    
    def __init__(self):
        self.cpu_usage = Gauge('system_cpu_usage_percent', 'CPU usage percentage')
        self.memory_usage = Gauge('system_memory_usage_percent', 'Memory usage percentage')
        self.disk_usage = Gauge('system_disk_usage_percent', 'Disk usage percentage')
        self.open_connections = Gauge('system_open_connections', 'Open network connections')
    
    def collect(self):
        """Collect current system metrics."""
        self.cpu_usage.set(psutil.cpu_percent())
        self.memory_usage.set(psutil.virtual_memory().percent)
        self.disk_usage.set(psutil.disk_usage('/').percent)
        self.open_connections.set(len(psutil.net_connections()))


# FastAPI Integration
from fastapi import FastAPI, Request
from starlette.middleware.base import BaseHTTPMiddleware

app = FastAPI()

class MetricsMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        ACTIVE_REQUESTS.inc()
        start_time = time.perf_counter()
        
        response = await call_next(request)
        
        duration = time.perf_counter() - start_time
        ACTIVE_REQUESTS.dec()
        
        REQUEST_COUNT.labels(
            method=request.method,
            endpoint=request.url.path,
            status=response.status_code
        ).inc()
        
        REQUEST_LATENCY.labels(
            method=request.method,
            endpoint=request.url.path
        ).observe(duration)
        
        return response

app.add_middleware(MetricsMiddleware)

@app.get("/metrics")
async def metrics():
    from prometheus_client import generate_latest
    return Response(generate_latest(), media_type="text/plain")

Profiling with py-spy and cProfile

import cProfile
import pstats
import io
from functools import wraps

def profile_function(func):
    """Decorator to profile function execution."""
    @wraps(func)
    def wrapper(*args, **kwargs):
        profiler = cProfile.Profile()
        profiler.enable()
        
        result = func(*args, **kwargs)
        
        profiler.disable()
        
        # Analyze results
        stream = io.StringIO()
        stats = pstats.Stats(profiler, stream=stream)
        stats.sort_stats('cumulative')
        stats.print_stats(20)  # Top 20 functions
        
        print(stream.getvalue())
        return result
    return wrapper

# Async profiling
import yappi

class AsyncProfiler:
    """Profile async code execution."""
    
    def __init__(self):
        self.started = False
    
    def start(self):
        yappi.set_clock_type("wall")
        yappi.start()
        self.started = True
    
    def stop(self):
        if self.started:
            yappi.stop()
    
    def get_stats(self):
        """Get profiling statistics."""
        func_stats = yappi.get_func_stats()
        thread_stats = yappi.get_thread_stats()
        
        print("\n=== Function Statistics ===")
        func_stats.sort("ttot", "desc").print_all()
        
        print("\n=== Thread Statistics ===")
        thread_stats.print_all()
        
        return func_stats, thread_stats
    
    def clear(self):
        yappi.clear_stats()

Database Query Analysis

import logging
from sqlalchemy import event
from sqlalchemy.engine import Engine
import time

# SQL Query Logger
logger = logging.getLogger('sqlalchemy.engine')
logger.setLevel(logging.INFO)

class QueryAnalyzer:
    """Analyze and log slow database queries."""
    
    def __init__(self, engine: Engine, slow_threshold: float = 0.1):
        self.engine = engine
        self.slow_threshold = slow_threshold
        self.query_stats = {}
        self._register_events()
    
    def _register_events(self):
        @event.listens_for(self.engine, "before_cursor_execute")
        def before_cursor_execute(
            conn, cursor, statement, parameters, context, executemany
        ):
            conn.info.setdefault('query_start_time', []).append(time.perf_counter())
        
        @event.listens_for(self.engine, "after_cursor_execute")
        def after_cursor_execute(
            conn, cursor, statement, parameters, context, executemany
        ):
            total_time = time.perf_counter() - conn.info['query_start_time'].pop()
            
            # Track query statistics
            query_key = statement[:100]  # First 100 chars as key
            if query_key not in self.query_stats:
                self.query_stats[query_key] = {
                    'count': 0,
                    'total_time': 0,
                    'max_time': 0
                }
            
            stats = self.query_stats[query_key]
            stats['count'] += 1
            stats['total_time'] += total_time
            stats['max_time'] = max(stats['max_time'], total_time)
            
            # Log slow queries
            if total_time > self.slow_threshold:
                logger.warning(
                    f"Slow query ({total_time:.3f}s): {statement[:200]}..."
                )
    
    def get_slow_queries(self, limit: int = 10):
        """Get top slow queries by total time."""
        sorted_queries = sorted(
            self.query_stats.items(),
            key=lambda x: x[1]['total_time'],
            reverse=True
        )
        return sorted_queries[:limit]
    
    def get_frequent_queries(self, limit: int = 10):
        """Get most frequently executed queries."""
        sorted_queries = sorted(
            self.query_stats.items(),
            key=lambda x: x[1]['count'],
            reverse=True
        )
        return sorted_queries[:limit]

Best Practices

  1. Establish baselines for normal system performance before investigating issues
  2. Monitor the four golden signals: latency, traffic, errors, and saturation
  3. Use percentiles (P95, P99) instead of averages for latency metrics
  4. Set up alerts for anomalies before users report issues
  5. Profile in production-like environments to get accurate results
  6. Focus on the critical path when optimizing performance
  7. Measure before and after any optimization to validate improvements
  8. Implement distributed tracing for microservices architectures
