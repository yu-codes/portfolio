Load balancing and distributed caching are fundamental techniques for building scalable, high-performance backend systems that can handle millions of concurrent requests.

Introduction

As applications grow, single servers cannot handle increasing traffic. Load balancing distributes incoming requests across multiple servers, while caching reduces database load by storing frequently accessed data in memory. Together, they form the backbone of scalable architectures.

NGINX Load Balancer Configuration

upstream backend_servers {
    # Weighted load balancing
    server backend1.example.com:8080 weight=3;
    server backend2.example.com:8080 weight=2;
    server backend3.example.com:8080 weight=1;
    
    # Health checks
    least_conn;
    keepalive 32;
}

server {
    listen 80;
    server_name api.example.com;
    
    location / {
        proxy_pass http://backend_servers;
        proxy_http_version 1.1;
        proxy_set_header Connection "";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        
        # Timeout settings
        proxy_connect_timeout 5s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
    }
    
    location /health {
        access_log off;
        return 200 "healthy";
    }
}

Redis Caching Implementation

import redis
import json
from functools import wraps
from typing import Optional, Callable, Any
import hashlib

class DistributedCache:
    def __init__(self, redis_url: str = "redis://localhost:6379"):
        self.redis = redis.from_url(
            redis_url,
            encoding="utf-8",
            decode_responses=True
        )
        self.default_ttl = 3600  # 1 hour
    
    def get(self, key: str) -> Optional[Any]:
        """Get value from cache with JSON deserialization."""
        value = self.redis.get(key)
        if value:
            return json.loads(value)
        return None
    
    def set(
        self, 
        key: str, 
        value: Any, 
        ttl: Optional[int] = None
    ) -> bool:
        """Set value in cache with JSON serialization."""
        return self.redis.setex(
            key,
            ttl or self.default_ttl,
            json.dumps(value)
        )
    
    def delete(self, key: str) -> bool:
        """Delete key from cache."""
        return bool(self.redis.delete(key))
    
    def invalidate_pattern(self, pattern: str) -> int:
        """Invalidate all keys matching pattern."""
        keys = self.redis.keys(pattern)
        if keys:
            return self.redis.delete(*keys)
        return 0
    
    def cached(
        self, 
        ttl: int = 3600, 
        key_prefix: str = ""
    ) -> Callable:
        """Decorator for caching function results."""
        def decorator(func: Callable) -> Callable:
            @wraps(func)
            def wrapper(*args, **kwargs) -> Any:
                # Generate cache key from function name and arguments
                key_data = f"{func.__name__}:{args}:{kwargs}"
                cache_key = f"{key_prefix}:{hashlib.md5(key_data.encode()).hexdigest()}"
                
                # Try to get from cache
                cached_value = self.get(cache_key)
                if cached_value is not None:
                    return cached_value
                
                # Execute function and cache result
                result = func(*args, **kwargs)
                self.set(cache_key, result, ttl)
                return result
            return wrapper
        return decorator

class CacheAsidePattern:
    """Implement cache-aside pattern for database queries."""
    
    def __init__(self, cache: DistributedCache, db_session):
        self.cache = cache
        self.db = db_session
    
    def get_user(self, user_id: int) -> Optional[dict]:
        cache_key = f"user:{user_id}"
        
        # 1. Check cache first
        user_data = self.cache.get(cache_key)
        if user_data:
            return user_data
        
        # 2. Cache miss - query database
        user = self.db.query(User).filter_by(id=user_id).first()
        if not user:
            return None
        
        user_data = user.to_dict()
        
        # 3. Update cache
        self.cache.set(cache_key, user_data, ttl=3600)
        
        return user_data
    
    def update_user(self, user_id: int, data: dict) -> bool:
        # 1. Update database
        user = self.db.query(User).filter_by(id=user_id).first()
        if not user:
            return False
        
        for key, value in data.items():
            setattr(user, key, value)
        self.db.commit()
        
        # 2. Invalidate cache
        self.cache.delete(f"user:{user_id}")
        
        return True

Cache Invalidation Strategies

from enum import Enum
from typing import List
import time

class InvalidationStrategy(Enum):
    TTL = "ttl"
    EVENT_DRIVEN = "event_driven"
    VERSION_BASED = "version_based"

class CacheInvalidator:
    def __init__(self, cache: DistributedCache):
        self.cache = cache
    
    def invalidate_by_ttl(self, key: str, ttl: int):
        """Let cache expire naturally via TTL."""
        # TTL is set when caching - no explicit invalidation needed
        pass
    
    def invalidate_by_event(self, entity: str, entity_id: int):
        """Invalidate cache when data changes."""
        patterns = [
            f"{entity}:{entity_id}",
            f"{entity}:list:*",
            f"{entity}:count"
        ]
        for pattern in patterns:
            self.cache.invalidate_pattern(pattern)
    
    def invalidate_by_version(self, key: str) -> str:
        """Use versioned keys for cache invalidation."""
        version = int(time.time() * 1000)
        return f"{key}:v{version}"
    
    def cascade_invalidate(self, entity: str, entity_id: int):
        """Invalidate related caches in cascade."""
        # Define relationships
        relationships = {
            "user": ["user_posts", "user_comments", "user_followers"],
            "post": ["post_comments", "feed", "trending"],
        }
        
        # Invalidate primary cache
        self.cache.delete(f"{entity}:{entity_id}")
        
        # Invalidate related caches
        for related in relationships.get(entity, []):
            self.cache.invalidate_pattern(f"{related}:*:{entity_id}:*")

Best Practices

  1. Choose the right load balancing algorithm based on your traffic patterns and server capabilities
  2. Implement health checks to quickly remove unhealthy servers from rotation
  3. Use connection pooling to reduce overhead of establishing new connections
  4. Set appropriate TTLs for cached data based on how frequently data changes
  5. Implement cache warming during deployments to prevent cold cache issues
  6. Monitor cache hit rates and adjust caching strategies accordingly
  7. Use consistent hashing for distributed caches to minimize redistribution on scaling
  8. Plan for cache failures with fallback mechanisms to the database
