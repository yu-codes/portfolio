Layered architecture separates concerns into distinct layers, improving maintainability, testability, and scalability of applications.

Traditional MVC Pattern

# Traditional MVC for web applications
from flask import Flask, request, jsonify, render_template

app = Flask(__name__)

# Model: Data and business logic
class User:
    def __init__(self, id: int, name: str, email: str):
        self.id = id
        self.name = name
        self.email = email

class UserRepository:
    def __init__(self, db):
        self.db = db
    
    def find_by_id(self, user_id: int) -> User:
        row = self.db.execute("SELECT * FROM users WHERE id = ?", [user_id])
        return User(**row) if row else None
    
    def save(self, user: User) -> User:
        self.db.execute(
            "INSERT INTO users (name, email) VALUES (?, ?)",
            [user.name, user.email]
        )
        return user

# Controller: Handles requests and coordinates
class UserController:
    def __init__(self, repository: UserRepository):
        self.repository = repository
    
    def get_user(self, user_id: int):
        user = self.repository.find_by_id(user_id)
        if not user:
            return None, 404
        return user, 200
    
    def create_user(self, data: dict):
        user = User(id=None, name=data['name'], email=data['email'])
        saved = self.repository.save(user)
        return saved, 201

# View: Presentation (Routes in this case)
@app.route('/users/<int:user_id>')
def get_user(user_id):
    user, status = controller.get_user(user_id)
    if status == 404:
        return jsonify({'error': 'Not found'}), 404
    return jsonify({'id': user.id, 'name': user.name})

Clean Architecture

# Domain Layer (innermost) - no external dependencies
from dataclasses import dataclass
from abc import ABC, abstractmethod
from typing import Optional, List

@dataclass
class User:
    """Domain entity - pure business logic."""
    id: Optional[str]
    email: str
    name: str
    
    def validate(self) -> bool:
        return '@' in self.email and len(self.name) > 0

class UserRepository(ABC):
    """Port - interface for data access."""
    
    @abstractmethod
    async def find_by_id(self, user_id: str) -> Optional[User]:
        pass
    
    @abstractmethod
    async def find_by_email(self, email: str) -> Optional[User]:
        pass
    
    @abstractmethod
    async def save(self, user: User) -> User:
        pass

class EmailService(ABC):
    """Port - interface for email service."""
    
    @abstractmethod
    async def send_welcome_email(self, user: User) -> bool:
        pass


# Application Layer - use cases
class CreateUserUseCase:
    """Application service - orchestrates business logic."""
    
    def __init__(
        self,
        user_repository: UserRepository,
        email_service: EmailService
    ):
        self.user_repository = user_repository
        self.email_service = email_service
    
    async def execute(self, email: str, name: str) -> User:
        # Business rule: check for existing email
        existing = await self.user_repository.find_by_email(email)
        if existing:
            raise UserAlreadyExistsError(email)
        
        # Create and validate domain entity
        user = User(id=None, email=email, name=name)
        if not user.validate():
            raise InvalidUserError("Invalid user data")
        
        # Persist
        saved_user = await self.user_repository.save(user)
        
        # Side effect: send welcome email
        await self.email_service.send_welcome_email(saved_user)
        
        return saved_user


# Infrastructure Layer - adapters
class PostgresUserRepository(UserRepository):
    """Adapter - implements port with PostgreSQL."""
    
    def __init__(self, pool):
        self.pool = pool
    
    async def find_by_id(self, user_id: str) -> Optional[User]:
        async with self.pool.acquire() as conn:
            row = await conn.fetchrow(
                "SELECT * FROM users WHERE id = $1", user_id
            )
            return User(**row) if row else None
    
    async def save(self, user: User) -> User:
        async with self.pool.acquire() as conn:
            row = await conn.fetchrow(
                "INSERT INTO users (email, name) VALUES ($1, $2) RETURNING *",
                user.email, user.name
            )
            return User(**row)

class SMTPEmailService(EmailService):
    """Adapter - implements port with SMTP."""
    
    async def send_welcome_email(self, user: User) -> bool:
        # SMTP implementation
        pass


# Interface Layer - controllers/handlers
from fastapi import FastAPI, Depends, HTTPException
from pydantic import BaseModel

app = FastAPI()

class CreateUserRequest(BaseModel):
    email: str
    name: str

class UserResponse(BaseModel):
    id: str
    email: str
    name: str

@app.post("/users", response_model=UserResponse)
async def create_user(
    request: CreateUserRequest,
    use_case: CreateUserUseCase = Depends(get_create_user_use_case)
):
    try:
        user = await use_case.execute(request.email, request.name)
        return UserResponse(id=user.id, email=user.email, name=user.name)
    except UserAlreadyExistsError:
        raise HTTPException(status_code=409, detail="Email already exists")

Dependency Injection

from dependency_injector import containers, providers

class Container(containers.DeclarativeContainer):
    config = providers.Configuration()
    
    # Infrastructure
    db_pool = providers.Singleton(
        create_db_pool,
        dsn=config.database_url
    )
    
    # Repositories
    user_repository = providers.Factory(
        PostgresUserRepository,
        pool=db_pool
    )
    
    # Services
    email_service = providers.Factory(
        SMTPEmailService,
        smtp_host=config.smtp_host
    )
    
    # Use Cases
    create_user_use_case = providers.Factory(
        CreateUserUseCase,
        user_repository=user_repository,
        email_service=email_service
    )

# Usage
container = Container()
container.config.from_yaml('config.yaml')

@app.post("/users")
@inject
async def create_user(
    request: CreateUserRequest,
    use_case: CreateUserUseCase = Depends(Provide[Container.create_user_use_case])
):
    return await use_case.execute(request.email, request.name)

Layer Dependencies

Clean Architecture Dependency Rule:

┌─────────────────────────────────────┐
│         Interface Layer             │  ← HTTP, CLI, Events
├─────────────────────────────────────┤
│        Application Layer            │  ← Use Cases
├─────────────────────────────────────┤
│          Domain Layer               │  ← Entities, Value Objects
├─────────────────────────────────────┤
│       Infrastructure Layer          │  ← DB, External APIs
└─────────────────────────────────────┘

Dependencies point INWARD only.
Inner layers know nothing about outer layers.

Best Practices

  1. Depend on Abstractions: Use interfaces/protocols, not concrete implementations
  2. Keep Domain Pure: No framework dependencies in domain layer
  3. Single Responsibility: Each layer has one reason to change
  4. Use Dependency Injection: Invert control for testability
  5. Test at Boundaries: Unit test domain, integration test adapters
  6. Start Simple: Add layers as complexity grows
