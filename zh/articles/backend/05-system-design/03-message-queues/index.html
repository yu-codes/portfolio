Apache Kafka

Producer and Consumer Implementation

from aiokafka import AIOKafkaProducer, AIOKafkaConsumer
from dataclasses import dataclass, asdict
import json
from typing import Callable, Dict, Any

@dataclass
class Event:
    event_type: str
    payload: Dict[str, Any]
    timestamp: str
    correlation_id: str

class KafkaProducer:
    def __init__(self, bootstrap_servers: str):
        self.bootstrap_servers = bootstrap_servers
        self.producer = None
    
    async def start(self):
        self.producer = AIOKafkaProducer(
            bootstrap_servers=self.bootstrap_servers,
            value_serializer=lambda v: json.dumps(v).encode(),
            acks='all',  # Wait for all replicas
            enable_idempotence=True,  # Exactly-once semantics
            max_batch_size=16384,
            linger_ms=10  # Batch for 10ms
        )
        await self.producer.start()
    
    async def stop(self):
        await self.producer.stop()
    
    async def send(
        self,
        topic: str,
        event: Event,
        key: str = None
    ):
        await self.producer.send_and_wait(
            topic,
            value=asdict(event),
            key=key.encode() if key else None
        )

class KafkaConsumer:
    def __init__(
        self,
        bootstrap_servers: str,
        group_id: str,
        topics: list[str]
    ):
        self.bootstrap_servers = bootstrap_servers
        self.group_id = group_id
        self.topics = topics
        self.handlers: Dict[str, Callable] = {}
    
    def register_handler(self, event_type: str, handler: Callable):
        self.handlers[event_type] = handler
    
    async def start(self):
        consumer = AIOKafkaConsumer(
            *self.topics,
            bootstrap_servers=self.bootstrap_servers,
            group_id=self.group_id,
            auto_offset_reset='earliest',
            enable_auto_commit=False,
            value_deserializer=lambda v: json.loads(v.decode())
        )
        await consumer.start()
        
        try:
            async for msg in consumer:
                event = msg.value
                handler = self.handlers.get(event['event_type'])
                
                if handler:
                    try:
                        await handler(event)
                        await consumer.commit()
                    except Exception as e:
                        # Send to dead letter queue
                        await self.send_to_dlq(msg, e)
                        await consumer.commit()
        finally:
            await consumer.stop()

class OrderAggregate:
    def __init__(self, order_id: str):
        self.order_id = order_id
        self.status = None
        self.items = []
        self.total = 0
        self.version = 0
    
    def apply(self, event: Event):
        if event.event_type == 'OrderCreated':
            self.status = 'created'
            self.items = event.payload['items']
            self.total = event.payload['total']
        elif event.event_type == 'OrderPaid':
            self.status = 'paid'
        elif event.event_type == 'OrderShipped':
            self.status = 'shipped'
        
        self.version += 1
    
    @classmethod
    def from_events(cls, order_id: str, events: list[Event]):
        aggregate = cls(order_id)
        for event in events:
            aggregate.apply(event)
        return aggregate

Advanced Patterns

import aio_pika
from aio_pika import ExchangeType, Message, DeliveryMode
import json

class RabbitMQClient:
    def __init__(self, url: str):
        self.url = url
        self.connection = None
        self.channel = None
    
    async def connect(self):
        self.connection = await aio_pika.connect_robust(self.url)
        self.channel = await self.connection.channel()
        await self.channel.set_qos(prefetch_count=10)
    
    async def setup_topology(self):
        # Topic exchange for event routing
        self.events_exchange = await self.channel.declare_exchange(
            'events', ExchangeType.TOPIC, durable=True
        )
        
        # Dead letter exchange
        self.dlx_exchange = await self.channel.declare_exchange(
            'dlx', ExchangeType.DIRECT, durable=True
        )
        
        # Queue with dead letter routing
        self.orders_queue = await self.channel.declare_queue(
            'order-events',
            durable=True,
            arguments={
                'x-dead-letter-exchange': 'dlx',
                'x-dead-letter-routing-key': 'order-events.dlq'
            }
        )
        
        # Bind queue to exchange
        await self.orders_queue.bind(self.events_exchange, 'order.*')
    
    async def publish(
        self,
        routing_key: str,
        payload: dict,
        correlation_id: str = None
    ):
        message = Message(
            body=json.dumps(payload).encode(),
            content_type='application/json',
            delivery_mode=DeliveryMode.PERSISTENT,
            correlation_id=correlation_id,
            headers={'version': '1.0'}
        )
        
        await self.events_exchange.publish(message, routing_key=routing_key)
    
    async def consume(self, queue_name: str, handler):
        queue = await self.channel.declare_queue(queue_name, durable=True)
        
        async def process_message(message: aio_pika.IncomingMessage):
            async with message.process(requeue=False):
                try:
                    payload = json.loads(message.body)
                    await handler(payload, message.correlation_id)
                except Exception as e:
                    # Message will go to DLQ due to x-dead-letter config
                    raise
        
        await queue.consume(process_message)

Retry with Exponential Backoff

class RetryHandler:
    MAX_RETRIES = 3
    RETRY_DELAYS = [1000, 5000, 30000]  # ms
    
    async def setup(self, channel):
        # Retry exchanges for each delay
        for delay in self.RETRY_DELAYS:
            exchange = await channel.declare_exchange(
                f'retry-{delay}', ExchangeType.DIRECT, durable=True
            )
            
            queue = await channel.declare_queue(
                f'retry-{delay}-queue',
                durable=True,
                arguments={
                    'x-dead-letter-exchange': 'events',
                    'x-message-ttl': delay
                }
            )
            await queue.bind(exchange, 'retry')
    
    async def retry_or_dlq(self, message, error, channel):
        retry_count = int(message.headers.get('x-retry-count', 0))
        
        if retry_count >= self.MAX_RETRIES:
            # Send to DLQ
            await self.dlx_exchange.publish(
                message, routing_key='failed'
            )
            return
        
        # Publish to retry queue
        delay = self.RETRY_DELAYS[retry_count]
        new_message = Message(
            body=message.body,
            headers={
                **message.headers,
                'x-retry-count': retry_count + 1,
                'x-last-error': str(error)
            }
        )
        
        retry_exchange = await channel.get_exchange(f'retry-{delay}')
        await retry_exchange.publish(new_message, routing_key='retry')

Event Design Patterns

# Event schema with versioning
from pydantic import BaseModel
from datetime import datetime
from typing import Literal

class EventBase(BaseModel):
    event_id: str
    event_type: str
    event_version: str = "1.0"
    timestamp: datetime
    correlation_id: str
    causation_id: str | None = None

class OrderCreatedV1(EventBase):
    event_type: Literal["order.created"] = "order.created"
    event_version: Literal["1.0"] = "1.0"
    
    class Payload(BaseModel):
        order_id: str
        user_id: str
        items: list[dict]
        total: float
    
    payload: Payload

Best Practices

  1. Idempotent Consumers: Handle duplicate messages gracefully
  2. Schema Evolution: Use versioned schemas with backward compatibility
  3. Dead Letter Queues: Capture and monitor failed messages
  4. Correlation IDs: Track requests across services
  5. Backpressure Handling: Implement rate limiting and circuit breakers
  6. Monitoring: Track queue depth, consumer lag, processing time
