from typing import Optional, Protocol
from dataclasses import dataclass, field
from collections.abc import Callable

# Use Protocol for structural typing
class Repository(Protocol):
    def find_by_id(self, id: int) -> Optional[dict]: ...
    def save(self, entity: dict) -> dict: ...

# Dataclasses for DTOs
@dataclass(frozen=True, slots=True)
class UserDTO:
    """Immutable user data transfer object."""
    id: int
    name: str
    email: str
    roles: list[str] = field(default_factory=list)

# Context managers for resource handling
class DatabaseConnection:
    def __enter__(self) -> "DatabaseConnection":
        self._connect()
        return self
    
    def __exit__(self, exc_type, exc_val, exc_tb) -> bool:
        self._disconnect()
        return False  # Don't suppress exceptions

# Type-safe function with docstring
def process_users(
    users: list[UserDTO],
    filter_fn: Callable[[UserDTO], bool] | None = None,
) -> list[UserDTO]:
    """
    Process and filter users.
    
    Args:
        users: List of users to process.
        filter_fn: Optional filter function.
    
    Returns:
        Filtered list of users.
    
    Raises:
        ValueError: If users list is empty.
    """
    if not users:
        raise ValueError("Users list cannot be empty")
    
    if filter_fn:
        return [u for u in users if filter_fn(u)]
    return users

Go: Effective Go Patterns

package users

import (
    "context"
    "errors"
    "fmt"
)

// Define errors at package level
var (
    ErrUserNotFound = errors.New("user not found")
    ErrInvalidInput = errors.New("invalid input")
)

// Interface segregation - small, focused interfaces
type UserReader interface {
    FindByID(ctx context.Context, id int64) (*User, error)
}

type UserWriter interface {
    Save(ctx context.Context, user *User) error
}

type UserRepository interface {
    UserReader
    UserWriter
}

// Functional options pattern
type ServiceOption func(*Service)

func WithTimeout(d time.Duration) ServiceOption {
    return func(s *Service) {
        s.timeout = d
    }
}

func WithCache(cache Cache) ServiceOption {
    return func(s *Service) {
        s.cache = cache
    }
}

func NewService(repo UserRepository, opts ...ServiceOption) *Service {
    s := &Service{
        repo:    repo,
        timeout: 30 * time.Second,
    }
    for _, opt := range opts {
        opt(s)
    }
    return s
}

// Error wrapping for context
func (s *Service) GetUser(ctx context.Context, id int64) (*User, error) {
    user, err := s.repo.FindByID(ctx, id)
    if err != nil {
        return nil, fmt.Errorf("GetUser(id=%d): %w", id, err)
    }
    return user, nil
}

// Use branded types for type safety
type UserId = number & { readonly brand: unique symbol };
type Email = string & { readonly brand: unique symbol };

function createUserId(id: number): UserId {
  return id as UserId;
}

// Result type for error handling
type Result<T, E = Error> =
  | { success: true; data: T }
  | { success: false; error: E };

function parseUser(json: unknown): Result<User> {
  try {
    // Validate and parse
    const user = userSchema.parse(json);
    return { success: true, data: user };
  } catch (error) {
    return { success: false, error: error as Error };
  }
}

// Discriminated unions for state
type AsyncState<T> =
  | { status: 'idle' }
  | { status: 'loading' }
  | { status: 'success'; data: T }
  | { status: 'error'; error: Error };

// Exhaustive switch checking
function renderState<T>(state: AsyncState<T>): string {
  switch (state.status) {
    case 'idle':
      return 'Ready';
    case 'loading':
      return 'Loading...';
    case 'success':
      return `Data: ${JSON.stringify(state.data)}`;
    case 'error':
      return `Error: ${state.error.message}`;
    default:
      const _exhaustive: never = state;
      return _exhaustive;
  }
}

Common Anti-Patterns to Avoid

# ❌ Mutable default arguments
def bad_function(items=[]):
    items.append(1)
    return items

# ✅ Use None as default
def good_function(items=None):
    if items is None:
        items = []
    items.append(1)
    return items

# ❌ Bare except clause
try:
    risky_operation()
except:
    pass

# ✅ Specific exception handling
try:
    risky_operation()
except ValueError as e:
    logger.error(f"Invalid value: {e}")
    raise
except Exception as e:
    logger.exception("Unexpected error")
    raise RuntimeError("Operation failed") from e

Best Practices

  1. Use Formatters: Automate formatting with pre-commit hooks
  2. Enable Strict Type Checking: Catch errors at compile time
  3. Follow SOLID Principles: Single responsibility, dependency injection
  4. Write Self-Documenting Code: Clear names, minimal comments
  5. Review Code Regularly: Enforce standards through code reviews
