Secure authentication is the gateway to protecting user accounts and sensitive data. This guide covers modern authentication methods, session management, and protection against common attacks.

Introduction

Authentication verifies user identity, while authorization determines what actions they can perform. A robust authentication system must protect against credential theft, session hijacking, and cross-site attacks while maintaining good user experience.

JWT Authentication Implementation

import jwt
from datetime import datetime, timedelta
from typing import Optional, Tuple
from pydantic import BaseModel
import secrets
import hashlib
from fastapi import HTTPException, Security, Depends
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials

class TokenConfig:
    SECRET_KEY = "your-secret-key-min-256-bits"  # Use env variable
    ALGORITHM = "HS256"
    ACCESS_TOKEN_EXPIRE_MINUTES = 15
    REFRESH_TOKEN_EXPIRE_DAYS = 7

class TokenPayload(BaseModel):
    sub: str  # User ID
    type: str  # access or refresh
    exp: datetime
    iat: datetime
    jti: str  # Unique token ID

class JWTAuthenticator:
    """Secure JWT-based authentication."""
    
    def __init__(self, config: TokenConfig):
        self.config = config
        self.revoked_tokens = set()  # Use Redis in production
    
    def create_access_token(self, user_id: str) -> str:
        """Create short-lived access token."""
        now = datetime.utcnow()
        payload = {
            "sub": user_id,
            "type": "access",
            "exp": now + timedelta(minutes=self.config.ACCESS_TOKEN_EXPIRE_MINUTES),
            "iat": now,
            "jti": secrets.token_hex(16)
        }
        return jwt.encode(
            payload,
            self.config.SECRET_KEY,
            algorithm=self.config.ALGORITHM
        )
    
    def create_refresh_token(self, user_id: str) -> str:
        """Create long-lived refresh token."""
        now = datetime.utcnow()
        payload = {
            "sub": user_id,
            "type": "refresh",
            "exp": now + timedelta(days=self.config.REFRESH_TOKEN_EXPIRE_DAYS),
            "iat": now,
            "jti": secrets.token_hex(16)
        }
        return jwt.encode(
            payload,
            self.config.SECRET_KEY,
            algorithm=self.config.ALGORITHM
        )
    
    def create_token_pair(self, user_id: str) -> Tuple[str, str]:
        """Create both access and refresh tokens."""
        return (
            self.create_access_token(user_id),
            self.create_refresh_token(user_id)
        )
    
    def verify_token(self, token: str, expected_type: str = "access") -> TokenPayload:
        """Verify and decode JWT token."""
        try:
            payload = jwt.decode(
                token,
                self.config.SECRET_KEY,
                algorithms=[self.config.ALGORITHM]
            )
            
            # Verify token type
            if payload.get("type") != expected_type:
                raise HTTPException(status_code=401, detail="Invalid token type")
            
            # Check if revoked
            if payload.get("jti") in self.revoked_tokens:
                raise HTTPException(status_code=401, detail="Token revoked")
            
            return TokenPayload(**payload)
            
        except jwt.ExpiredSignatureError:
            raise HTTPException(status_code=401, detail="Token expired")
        except jwt.InvalidTokenError:
            raise HTTPException(status_code=401, detail="Invalid token")
    
    def revoke_token(self, token: str):
        """Revoke a token by adding JTI to blacklist."""
        try:
            payload = jwt.decode(
                token,
                self.config.SECRET_KEY,
                algorithms=[self.config.ALGORITHM],
                options={"verify_exp": False}
            )
            self.revoked_tokens.add(payload["jti"])
        except jwt.InvalidTokenError:
            pass
    
    def refresh_access_token(self, refresh_token: str) -> str:
        """Get new access token using refresh token."""
        payload = self.verify_token(refresh_token, expected_type="refresh")
        return self.create_access_token(payload.sub)

Session-Based Authentication

import secrets
import hashlib
from datetime import datetime, timedelta
from typing import Optional
import redis
from fastapi import Request, Response, HTTPException

class SecureSessionManager:
    """Secure server-side session management."""
    
    def __init__(self, redis_client: redis.Redis):
        self.redis = redis_client
        self.session_ttl = 3600  # 1 hour
        self.cookie_name = "session_id"
    
    def create_session(
        self, 
        user_id: str, 
        response: Response,
        user_data: dict = None
    ) -> str:
        """Create new session with secure cookie."""
        # Generate cryptographically secure session ID
        session_id = secrets.token_urlsafe(32)
        
        # Store session data in Redis
        session_data = {
            "user_id": user_id,
            "created_at": datetime.utcnow().isoformat(),
            "ip_address": "",  # Set from request
            "user_agent": "",  # Set from request
            **(user_data or {})
        }
        
        self.redis.hset(
            f"session:{session_id}",
            mapping=session_data
        )
        self.redis.expire(f"session:{session_id}", self.session_ttl)
        
        # Set secure cookie
        response.set_cookie(
            key=self.cookie_name,
            value=session_id,
            httponly=True,     # Not accessible via JavaScript
            secure=True,       # HTTPS only
            samesite="lax",    # CSRF protection
            max_age=self.session_ttl
        )
        
        return session_id
    
    def validate_session(
        self, 
        request: Request
    ) -> Optional[dict]:
        """Validate session from request cookie."""
        session_id = request.cookies.get(self.cookie_name)
        if not session_id:
            return None
        
        # Get session from Redis
        session_data = self.redis.hgetall(f"session:{session_id}")
        if not session_data:
            return None
        
        # Validate session binding (optional additional security)
        # if session_data.get('ip_address') != request.client.host:
        #     self.destroy_session(session_id)
        #     return None
        
        # Refresh TTL on activity
        self.redis.expire(f"session:{session_id}", self.session_ttl)
        
        return session_data
    
    def destroy_session(self, session_id: str, response: Response = None):
        """Destroy session on logout."""
        self.redis.delete(f"session:{session_id}")
        
        if response:
            response.delete_cookie(
                key=self.cookie_name,
                httponly=True,
                secure=True,
                samesite="lax"
            )
    
    def destroy_all_user_sessions(self, user_id: str):
        """Destroy all sessions for a user (password change, etc.)."""
        pattern = "session:*"
        for key in self.redis.scan_iter(pattern):
            session_data = self.redis.hgetall(key)
            if session_data.get(b'user_id', b'').decode() == user_id:
                self.redis.delete(key)

CSRF Protection

import secrets
import hmac
import hashlib
from fastapi import Request, HTTPException
from typing import Optional

class CSRFProtection:
    """CSRF protection using double-submit cookie pattern."""
    
    def __init__(self, secret_key: str):
        self.secret_key = secret_key
        self.cookie_name = "csrf_token"
        self.header_name = "X-CSRF-Token"
    
    def generate_token(self, session_id: str) -> str:
        """Generate CSRF token bound to session."""
        random_value = secrets.token_hex(16)
        signature = hmac.new(
            self.secret_key.encode(),
            f"{session_id}:{random_value}".encode(),
            hashlib.sha256
        ).hexdigest()
        return f"{random_value}.{signature}"
    
    def verify_token(
        self, 
        request: Request, 
        session_id: str
    ) -> bool:
        """Verify CSRF token from header matches cookie."""
        # Get token from cookie
        cookie_token = request.cookies.get(self.cookie_name)
        # Get token from header (or form field)
        header_token = request.headers.get(self.header_name)
        
        if not cookie_token or not header_token:
            raise HTTPException(status_code=403, detail="CSRF token missing")
        
        if cookie_token != header_token:
            raise HTTPException(status_code=403, detail="CSRF token mismatch")
        
        # Verify signature
        try:
            random_value, signature = cookie_token.split('.')
            expected_signature = hmac.new(
                self.secret_key.encode(),
                f"{session_id}:{random_value}".encode(),
                hashlib.sha256
            ).hexdigest()
            
            if not hmac.compare_digest(signature, expected_signature):
                raise HTTPException(status_code=403, detail="Invalid CSRF token")
        except ValueError:
            raise HTTPException(status_code=403, detail="Malformed CSRF token")
        
        return True


# Middleware for CSRF protection
@app.middleware("http")
async def csrf_middleware(request: Request, call_next):
    # Skip CSRF for safe methods
    if request.method in ("GET", "HEAD", "OPTIONS"):
        return await call_next(request)
    
    # Skip CSRF for API with bearer tokens
    if request.headers.get("Authorization", "").startswith("Bearer"):
        return await call_next(request)
    
    # Verify CSRF for session-based auth
    session_id = request.cookies.get("session_id")
    if session_id:
        csrf_protection.verify_token(request, session_id)
    
    return await call_next(request)

Multi-Factor Authentication

import pyotp
import qrcode
import io
import base64

class MFAManager:
    """Time-based One-Time Password (TOTP) MFA."""
    
    def __init__(self, issuer: str = "MyApp"):
        self.issuer = issuer
    
    def generate_secret(self) -> str:
        """Generate TOTP secret for user."""
        return pyotp.random_base32()
    
    def get_provisioning_uri(self, secret: str, email: str) -> str:
        """Get URI for QR code."""
        totp = pyotp.TOTP(secret)
        return totp.provisioning_uri(
            name=email,
            issuer_name=self.issuer
        )
    
    def generate_qr_code(self, secret: str, email: str) -> str:
        """Generate QR code as base64 image."""
        uri = self.get_provisioning_uri(secret, email)
        qr = qrcode.make(uri)
        
        buffer = io.BytesIO()
        qr.save(buffer, format='PNG')
        return base64.b64encode(buffer.getvalue()).decode()
    
    def verify_code(self, secret: str, code: str) -> bool:
        """Verify TOTP code with time window."""
        totp = pyotp.TOTP(secret)
        return totp.verify(code, valid_window=1)  # Allow 30 second window
    
    def generate_backup_codes(self, count: int = 10) -> list[str]:
        """Generate one-time backup codes."""
        return [secrets.token_hex(4).upper() for _ in range(count)]

Best Practices

  1. Use HTTPS everywhere and set secure cookie flags
  2. Implement rate limiting on authentication endpoints
  3. Use strong password policies with Argon2id hashing
  4. Set appropriate token lifetimes (short for access, longer for refresh)
  5. Implement token revocation for logout and security events
  6. Use CSRF protection for session-based authentication
  7. Add MFA option for sensitive accounts
  8. Log authentication events for security monitoring
