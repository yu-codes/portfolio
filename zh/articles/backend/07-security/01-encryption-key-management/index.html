Encryption & Key Management

Data encryption strategies and secure key management practices.

Encryption and key management form the foundation of data security, protecting sensitive information at rest and in transit while ensuring only authorized parties can access decrypted data.

Introduction

Proper encryption protects data confidentiality even when other security controls fail. Key management is equally critical—poor key handling can render even the strongest encryption useless. This guide covers practical encryption patterns and secure key management strategies.

Encryption Types Comparison

TypeUse CaseSpeedKey Management

Symmetric Encryption Implementation

AES-256-GCM Encryption

from cryptography.hazmat.primitives.ciphers.aead import AESGCM
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
import os
import base64
from typing import Tuple

class AESEncryption:
    """AES-256-GCM encryption for data at rest."""
    
    def __init__(self, key: bytes):
        if len(key) != 32:  # 256 bits
            raise ValueError("Key must be 32 bytes (256 bits)")
        self.key = key
        self.aesgcm = AESGCM(key)
    
    def encrypt(self, plaintext: bytes, associated_data: bytes = None) -> bytes:
        """
        Encrypt data with AES-256-GCM.
        Returns: nonce (12 bytes) + ciphertext + tag (16 bytes)
        """
        nonce = os.urandom(12)  # 96-bit nonce for GCM
        ciphertext = self.aesgcm.encrypt(nonce, plaintext, associated_data)
        return nonce + ciphertext
    
    def decrypt(self, ciphertext: bytes, associated_data: bytes = None) -> bytes:
        """Decrypt AES-256-GCM encrypted data."""
        nonce = ciphertext[:12]
        actual_ciphertext = ciphertext[12:]
        return self.aesgcm.decrypt(nonce, actual_ciphertext, associated_data)
    
    @staticmethod
    def derive_key_from_password(
        password: str, 
        salt: bytes = None
    ) -> Tuple[bytes, bytes]:
        """Derive encryption key from password using PBKDF2."""
        if salt is None:
            salt = os.urandom(16)
        
        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=32,
            salt=salt,
            iterations=480000,  # OWASP recommended minimum
        )
        key = kdf.derive(password.encode())
        return key, salt


class FieldLevelEncryption:
    """Encrypt specific fields in database records."""
    
    def __init__(self, encryption: AESEncryption):
        self.encryption = encryption
        self.encrypted_fields = {'ssn', 'credit_card', 'bank_account'}
    
    def encrypt_record(self, record: dict) -> dict:
        """Encrypt sensitive fields in a record."""
        encrypted = record.copy()
        for field in self.encrypted_fields:
            if field in encrypted and encrypted[field]:
                plaintext = str(encrypted[field]).encode()
                ciphertext = self.encryption.encrypt(plaintext)
                encrypted[field] = base64.b64encode(ciphertext).decode()
        return encrypted
    
    def decrypt_record(self, record: dict) -> dict:
        """Decrypt sensitive fields in a record."""
        decrypted = record.copy()
        for field in self.encrypted_fields:
            if field in decrypted and decrypted[field]:
                ciphertext = base64.b64decode(decrypted[field])
                plaintext = self.encryption.decrypt(ciphertext)
                decrypted[field] = plaintext.decode()
        return decrypted


# Usage
key = os.urandom(32)  # In production, use key management service
encryption = AESEncryption(key)

field_encryption = FieldLevelEncryption(encryption)

user_data = {
    'name': 'John Doe',
    'email': 'john@example.com',
    'ssn': '123-45-6789',
    'credit_card': '4111111111111111'
}

encrypted_data = field_encryption.encrypt_record(user_data)
print(encrypted_data)
# {'name': 'John Doe', 'email': 'john@example.com', 
#  'ssn': 'BASE64_ENCRYPTED', 'credit_card': 'BASE64_ENCRYPTED'}

Key Management with AWS KMS

import boto3
import base64
from botocore.exceptions import ClientError
from functools import lru_cache

class KMSKeyManager:
    """AWS KMS integration for key management."""
    
    def __init__(self, key_id: str, region: str = 'us-east-1'):
        self.kms = boto3.client('kms', region_name=region)
        self.key_id = key_id
        self._data_key_cache = {}
    
    def generate_data_key(self) -> Tuple[bytes, bytes]:
        """
        Generate a data encryption key (DEK) using envelope encryption.
        Returns (plaintext_key, encrypted_key)
        """
        response = self.kms.generate_data_key(
            KeyId=self.key_id,
            KeySpec='AES_256'
        )
        return (
            response['Plaintext'],
            response['CiphertextBlob']
        )
    
    def decrypt_data_key(self, encrypted_key: bytes) -> bytes:
        """Decrypt a data encryption key."""
        response = self.kms.decrypt(
            KeyId=self.key_id,
            CiphertextBlob=encrypted_key
        )
        return response['Plaintext']
    
    def encrypt_with_envelope(
        self, 
        plaintext: bytes
    ) -> dict:
        """
        Envelope encryption: encrypt data with DEK, encrypt DEK with KMS.
        """
        # Generate new data key
        plaintext_key, encrypted_key = self.generate_data_key()
        
        # Encrypt data with plaintext key
        aesgcm = AESGCM(plaintext_key)
        nonce = os.urandom(12)
        ciphertext = aesgcm.encrypt(nonce, plaintext, None)
        
        # Clear plaintext key from memory
        del plaintext_key
        
        return {
            'encrypted_key': base64.b64encode(encrypted_key).decode(),
            'nonce': base64.b64encode(nonce).decode(),
            'ciphertext': base64.b64encode(ciphertext).decode()
        }
    
    def decrypt_with_envelope(self, envelope: dict) -> bytes:
        """Decrypt envelope-encrypted data."""
        encrypted_key = base64.b64decode(envelope['encrypted_key'])
        nonce = base64.b64decode(envelope['nonce'])
        ciphertext = base64.b64decode(envelope['ciphertext'])
        
        # Decrypt data key with KMS
        plaintext_key = self.decrypt_data_key(encrypted_key)
        
        # Decrypt data with plaintext key
        aesgcm = AESGCM(plaintext_key)
        plaintext = aesgcm.decrypt(nonce, ciphertext, None)
        
        # Clear plaintext key from memory
        del plaintext_key
        
        return plaintext

HashiCorp Vault Integration

import hvac
from typing import Optional
import os

class VaultKeyManager:
    """HashiCorp Vault integration for secrets management."""
    
    def __init__(self, url: str = None, token: str = None):
        self.client = hvac.Client(
            url=url or os.environ.get('VAULT_ADDR'),
            token=token or os.environ.get('VAULT_TOKEN')
        )
    
    def store_secret(self, path: str, data: dict) -> bool:
        """Store a secret in Vault."""
        self.client.secrets.kv.v2.create_or_update_secret(
            path=path,
            secret=data
        )
        return True
    
    def get_secret(self, path: str) -> Optional[dict]:
        """Retrieve a secret from Vault."""
        try:
            response = self.client.secrets.kv.v2.read_secret_version(
                path=path
            )
            return response['data']['data']
        except hvac.exceptions.InvalidPath:
            return None
    
    def encrypt_transit(self, plaintext: str, key_name: str) -> str:
        """Encrypt using Vault Transit secrets engine."""
        response = self.client.secrets.transit.encrypt_data(
            name=key_name,
            plaintext=base64.b64encode(plaintext.encode()).decode()
        )
        return response['data']['ciphertext']
    
    def decrypt_transit(self, ciphertext: str, key_name: str) -> str:
        """Decrypt using Vault Transit secrets engine."""
        response = self.client.secrets.transit.decrypt_data(
            name=key_name,
            ciphertext=ciphertext
        )
        return base64.b64decode(
            response['data']['plaintext']
        ).decode()
    
    def generate_database_credentials(self, role: str) -> dict:
        """Generate dynamic database credentials."""
        response = self.client.secrets.database.generate_credentials(
            name=role
        )
        return {
            'username': response['data']['username'],
            'password': response['data']['password'],
            'lease_id': response['lease_id'],
            'lease_duration': response['lease_duration']
        }

Password Hashing

import argon2
from argon2 import PasswordHasher
from argon2.exceptions import VerifyMismatchError

class SecurePasswordHandler:
    """Secure password hashing with Argon2id."""
    
    def __init__(self):
        # Argon2id with recommended parameters
        self.hasher = PasswordHasher(
            time_cost=3,        # iterations
            memory_cost=65536,  # 64MB
            parallelism=4,      # threads
            hash_len=32,
            salt_len=16
        )
    
    def hash_password(self, password: str) -> str:
        """Hash password with Argon2id."""
        return self.hasher.hash(password)
    
    def verify_password(self, password: str, hash: str) -> bool:
        """Verify password against hash."""
        try:
            self.hasher.verify(hash, password)
            return True
        except VerifyMismatchError:
            return False
    
    def needs_rehash(self, hash: str) -> bool:
        """Check if hash needs to be upgraded."""
        return self.hasher.check_needs_rehash(hash)


# Usage
password_handler = SecurePasswordHandler()

# Hash password for storage
hashed = password_handler.hash_password("user_password")
# $argon2id$v=19$m=65536,t=3,p=4$...

# Verify password
is_valid = password_handler.verify_password("user_password", hashed)

Key Rotation Strategy

StrategyDescriptionUse Case
Automatic rotationKMS rotates keys on scheduleLow maintenance
Manual rotationTriggered by eventsSecurity incidents
Re-encryptionDecrypt and re-encrypt dataCompliance requirements
Key versioningMultiple key versions activeGradual migration

Best Practices

  1. Never hardcode keys in source code or configuration files
  2. Use envelope encryption for large data sets
  3. Enable automatic key rotation where supported
  4. Implement key versioning for seamless rotation
  5. Use authenticated encryption (GCM mode) to prevent tampering
  6. Store encrypted keys separate from encrypted data
  7. Audit key access and usage regularly
  8. Have a key recovery plan for disaster scenarios
