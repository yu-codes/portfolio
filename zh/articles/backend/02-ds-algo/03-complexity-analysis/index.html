Common Time Complexities

from typing import List

# O(1) - Constant Time
def get_first(arr: List[int]) -> int:
    """Same time regardless of input size."""
    return arr[0] if arr else None

# O(log n) - Logarithmic Time
def binary_search(arr: List[int], target: int) -> int:
    """Each step halves the search space."""
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1

# O(n) - Linear Time
def find_max(arr: List[int]) -> int:
    """Must examine each element once."""
    max_val = arr[0]
    for num in arr:
        if num > max_val:
            max_val = num
    return max_val

# O(n log n) - Linearithmic Time
def merge_sort(arr: List[int]) -> List[int]:
    """Divide and conquer with merge."""
    if len(arr) <= 1:
        return arr
    mid = len(arr) // 2
    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])
    return merge(left, right)

# O(n²) - Quadratic Time
def bubble_sort(arr: List[int]) -> List[int]:
    """Nested loops over input."""
    n = len(arr)
    for i in range(n):
        for j in range(n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
    return arr

# O(2^n) - Exponential Time
def fibonacci_naive(n: int) -> int:
    """Recursive without memoization."""
    if n <= 1:
        return n
    return fibonacci_naive(n - 1) + fibonacci_naive(n - 2)

# O(1) Space - In-place
def reverse_inplace(arr: List[int]) -> None:
    """Only uses constant extra space."""
    left, right = 0, len(arr) - 1
    while left < right:
        arr[left], arr[right] = arr[right], arr[left]
        left += 1
        right -= 1

# O(n) Space - Linear
def reverse_copy(arr: List[int]) -> List[int]:
    """Creates a new array of same size."""
    return arr[::-1]

def two_sum(nums: List[int], target: int) -> List[int]:
    """Uses hash map for O(1) lookups."""
    seen = {}
    for i, num in enumerate(nums):
        complement = target - num
        if complement in seen:
            return [seen[complement], i]
        seen[num] = i
    return []

Amortized Analysis

class DynamicArray:
    """
    Amortized O(1) append despite occasional O(n) resizing.
    
    Analysis:
    - n appends cost: n + (1 + 2 + 4 + ... + n) ≈ 3n
    - Average cost per append: O(3n/n) = O(1)
    """
    
    def __init__(self):
        self._capacity = 1
        self._size = 0
        self._data = [None] * self._capacity
    
    def append(self, value):
        if self._size == self._capacity:
            # O(n) resize operation
            self._resize(self._capacity * 2)
        self._data[self._size] = value  # O(1)
        self._size += 1
    
    def _resize(self, new_capacity):
        new_data = [None] * new_capacity
        for i in range(self._size):
            new_data[i] = self._data[i]
        self._data = new_data
        self._capacity = new_capacity

def complex_analysis(n: int, m: int) -> int:
    """
    Time Complexity Analysis:
    - Loop 1: O(n)
    - Loop 2 (nested): O(n * m)
    - Loop 3: O(m log m) [sorting]
    
    Total: O(n + nm + m log m) = O(nm) when n, m are similar
    
    Space Complexity:
    - result list: O(n)
    - sorted_data: O(m)
    
    Total: O(n + m)
    """
    
    # O(n) time, O(n) space

    total = 0
    for i in range(n):
        for j in range(m):
            total += i * j
    
    # O(m log m) time, O(m) space
    data = list(range(m))
    sorted_data = sorted(data)
    
    return total

Best Practices for Optimization

# Before optimization: O(n²)
def find_pairs_slow(arr: List[int], target: int) -> List[tuple]:
    pairs = []
    for i in range(len(arr)):
        for j in range(i + 1, len(arr)):
            if arr[i] + arr[j] == target:
                pairs.append((arr[i], arr[j]))
    return pairs

# After optimization: O(n)
def find_pairs_fast(arr: List[int], target: int) -> List[tuple]:
    pairs = []
    seen = set()
    for num in arr:
        complement = target - num
        if complement in seen:
            pairs.append((complement, num))
        seen.add(num)
    return pairs

Best Practices

  1. Focus on Dominant Terms: O(n² + n) = O(n²)
  2. Consider Average vs Worst Case: QuickSort O(n log n) avg, O(n²) worst
  3. Space-Time Trade-offs: Hash tables use O(n) space for O(1) lookups
  4. Profile Real Performance: Big O doesn’t capture constants or cache behavior
  5. Optimize Hot Paths: Focus on code executed most frequently
