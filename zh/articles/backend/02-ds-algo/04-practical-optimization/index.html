Database Query Optimization

From N+1 to Batch Processing

# ❌ N+1 Query Problem - O(n) database calls
async def get_users_slow(user_ids: list[int]) -> list[User]:
    users = []
    for user_id in user_ids:  # N iterations
        user = await db.execute(
            "SELECT * FROM users WHERE id = ?", [user_id]
        )  # 1 query each
        users.append(user)
    return users

# ✅ Batch Query - O(1) database call
async def get_users_fast(user_ids: list[int]) -> list[User]:
    if not user_ids:
        return []
    
    placeholders = ','.join(['?'] * len(user_ids))
    users = await db.execute(
        f"SELECT * FROM users WHERE id IN ({placeholders})",
        user_ids
    )
    return users

# ✅ With indexing consideration
# CREATE INDEX idx_users_id ON users(id);  -- B-tree index
# Query complexity: O(k log n) where k = len(user_ids)

Pagination Optimization

# ❌ Offset-based pagination - O(offset + limit)
async def get_page_slow(page: int, page_size: int):
    offset = page * page_size
    # Database must scan offset + limit rows
    return await db.execute(
        "SELECT * FROM posts ORDER BY created_at DESC "
        "LIMIT ? OFFSET ?",
        [page_size, offset]
    )

# ✅ Cursor-based pagination - O(limit)
async def get_page_fast(cursor: str | None, page_size: int):
    if cursor:
        return await db.execute(
            "SELECT * FROM posts "
            "WHERE created_at < ? "
            "ORDER BY created_at DESC LIMIT ?",
            [cursor, page_size]
        )
    return await db.execute(
        "SELECT * FROM posts ORDER BY created_at DESC LIMIT ?",
        [page_size]
    )

Caching Strategies

LRU Cache Implementation

from collections import OrderedDict
from typing import Generic, TypeVar, Optional

K = TypeVar('K')
V = TypeVar('V')

class LRUCache(Generic[K, V]):
    """O(1) get and put operations."""
    
    def __init__(self, capacity: int):
        self.capacity = capacity
        self.cache: OrderedDict[K, V] = OrderedDict()
    
    def get(self, key: K) -> Optional[V]:
        if key not in self.cache:
            return None
        # Move to end (most recently used)
        self.cache.move_to_end(key)
        return self.cache[key]
    
    def put(self, key: K, value: V) -> None:
        if key in self.cache:
            self.cache.move_to_end(key)
        self.cache[key] = value
        if len(self.cache) > self.capacity:
            # Remove least recently used
            self.cache.popitem(last=False)

# Application: API Response Caching
class CachedUserService:
    def __init__(self, db, cache_size: int = 1000):
        self.db = db
        self.cache = LRUCache[int, User](cache_size)
    
    async def get_user(self, user_id: int) -> User:
        # Check cache first - O(1)
        cached = self.cache.get(user_id)
        if cached:
            return cached
        
        # Fetch from DB - O(log n) with index
        user = await self.db.get_user(user_id)
        self.cache.put(user_id, user)
        return user

Rate Limiting with Token Bucket

import time
from dataclasses import dataclass

@dataclass
class TokenBucket:
    """Rate limiter with O(1) check operation."""
    capacity: int
    refill_rate: float  # tokens per second
    tokens: float = 0.0
    last_refill: float = 0.0
    
    def __post_init__(self):
        self.tokens = float(self.capacity)
        self.last_refill = time.time()
    
    def allow_request(self, tokens_needed: int = 1) -> bool:
        self._refill()
        
        if self.tokens >= tokens_needed:
            self.tokens -= tokens_needed
            return True
        return False
    
    def _refill(self):
        now = time.time()
        elapsed = now - self.last_refill
        self.tokens = min(
            self.capacity,
            self.tokens + elapsed * self.refill_rate
        )
        self.last_refill = now

# Sliding window rate limiter
from collections import deque

class SlidingWindowRateLimiter:
    """More precise rate limiting with O(1) amortized."""
    
    def __init__(self, max_requests: int, window_seconds: float):
        self.max_requests = max_requests
        self.window_seconds = window_seconds
        self.requests: deque[float] = deque()
    
    def allow_request(self) -> bool:
        now = time.time()
        window_start = now - self.window_seconds
        
        # Remove old requests - amortized O(1)
        while self.requests and self.requests[0] < window_start:
            self.requests.popleft()
        
        if len(self.requests) < self.max_requests:
            self.requests.append(now)
            return True
        return False

String Processing Optimization

# ❌ String concatenation in loop - O(n²)
def build_string_slow(items: list[str]) -> str:
    result = ""
    for item in items:
        result += item  # Creates new string each time
    return result

# ✅ Using join - O(n)
def build_string_fast(items: list[str]) -> str:
    return "".join(items)

# ✅ Using StringIO for complex building
from io import StringIO

def build_complex_string(data: list[dict]) -> str:
    buffer = StringIO()
    for item in data:
        buffer.write(f"{item['name']}: {item['value']}\n")
    return buffer.getvalue()

Profiling and Measurement

import cProfile
import pstats
from functools import wraps
import time

def profile(func):
    """Decorator for function profiling."""
    @wraps(func)
    def wrapper(*args, **kwargs):
        profiler = cProfile.Profile()
        profiler.enable()
        result = func(*args, **kwargs)
        profiler.disable()
        
        stats = pstats.Stats(profiler)
        stats.sort_stats('cumulative')
        stats.print_stats(10)  # Top 10 functions
        
        return result
    return wrapper

def timeit(func):
    """Simple timing decorator."""
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"{func.__name__} took {elapsed:.4f}s")
        return result
    return wrapper

Best Practices

  1. Measure First: Profile before optimizing to find actual bottlenecks
  2. Batch Operations: Reduce I/O calls by grouping operations
  3. Use Appropriate Data Structures: HashMap for lookups, sorted structures for range queries
  4. Cache Strategically: Cache expensive computations with proper invalidation
  5. Consider Memory: Sometimes O(n) space for O(1) time is worthwhile
