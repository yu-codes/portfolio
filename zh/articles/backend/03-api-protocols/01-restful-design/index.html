RESTful Design & Versioning

Designing RESTful APIs with proper versioning strategies and best practices.

RESTful API design provides a standardized approach to building web services that are scalable, maintainable, and intuitive to use.

REST Principles

Resource-Oriented Design

# Resources are nouns, not verbs
✅ GET /users/123          # Get user
✅ POST /users             # Create user
✅ PUT /users/123          # Update user
✅ DELETE /users/123       # Delete user

❌ GET /getUser?id=123
❌ POST /createUser
❌ POST /deleteUser

# Nested resources for relationships
GET /users/123/orders              # User's orders
GET /users/123/orders/456          # Specific order
POST /users/123/orders             # Create order for user

# Query parameters for filtering/pagination
GET /users?status=active&role=admin
GET /orders?created_after=2024-01-01&limit=20&offset=40

HTTP Methods and Status Codes

from fastapi import FastAPI, HTTPException, status
from pydantic import BaseModel
from typing import Optional

app = FastAPI()

class UserCreate(BaseModel):
    name: str
    email: str

class UserUpdate(BaseModel):
    name: Optional[str] = None
    email: Optional[str] = None

class UserResponse(BaseModel):
    id: int
    name: str
    email: str

# GET - Retrieve resource (idempotent)
@app.get("/users/{user_id}", response_model=UserResponse)
async def get_user(user_id: int):
    user = await db.find_user(user_id)
    if not user:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail="User not found"
        )
    return user

# POST - Create resource (not idempotent)
@app.post("/users", response_model=UserResponse, status_code=status.HTTP_201_CREATED)
async def create_user(user: UserCreate):
    if await db.email_exists(user.email):
        raise HTTPException(
            status_code=status.HTTP_409_CONFLICT,
            detail="Email already exists"
        )
    return await db.create_user(user)

# PUT - Full update (idempotent)
@app.put("/users/{user_id}", response_model=UserResponse)
async def update_user(user_id: int, user: UserCreate):
    updated = await db.update_user(user_id, user)
    if not updated:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND)
    return updated

# PATCH - Partial update
@app.patch("/users/{user_id}", response_model=UserResponse)
async def patch_user(user_id: int, user: UserUpdate):
    updated = await db.patch_user(user_id, user.dict(exclude_unset=True))
    if not updated:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND)
    return updated

# DELETE - Remove resource (idempotent)
@app.delete("/users/{user_id}", status_code=status.HTTP_204_NO_CONTENT)
async def delete_user(user_id: int):
    deleted = await db.delete_user(user_id)
    if not deleted:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND)

API Versioning Strategies

URL Path Versioning

from fastapi import APIRouter

# Version in URL path - most explicit
v1_router = APIRouter(prefix="/api/v1")
v2_router = APIRouter(prefix="/api/v2")

@v1_router.get("/users/{user_id}")
async def get_user_v1(user_id: int):
    # Original implementation
    return {"id": user_id, "name": "..."}

@v2_router.get("/users/{user_id}")
async def get_user_v2(user_id: int):
    # New implementation with additional fields
    return {
        "id": user_id,
        "name": "...",
        "metadata": {"version": 2}
    }

app.include_router(v1_router)
app.include_router(v2_router)

Header-Based Versioning

from fastapi import Header, Depends

async def get_api_version(
    api_version: str = Header(default="1", alias="X-API-Version")
) -> int:
    return int(api_version)

@app.get("/users/{user_id}")
async def get_user(user_id: int, version: int = Depends(get_api_version)):
    if version == 2:
        return await get_user_v2(user_id)
    return await get_user_v1(user_id)

Content Negotiation

from fastapi import Header

@app.get("/users/{user_id}")
async def get_user(
    user_id: int,
    accept: str = Header(default="application/vnd.api.v1+json")
):
    if "v2" in accept:
        return await get_user_v2(user_id)
    return await get_user_v1(user_id)

Response Design

from pydantic import BaseModel
from typing import Generic, TypeVar, List, Optional
from datetime import datetime

T = TypeVar('T')

# Consistent response envelope
class ApiResponse(BaseModel, Generic[T]):
    success: bool = True
    data: T
    timestamp: datetime = None
    
    def __init__(self, **data):
        super().__init__(**data)
        self.timestamp = datetime.utcnow()

# Paginated response
class PaginatedResponse(BaseModel, Generic[T]):
    items: List[T]
    total: int
    page: int
    page_size: int
    has_next: bool
    has_prev: bool

# Error response
class ErrorResponse(BaseModel):
    success: bool = False
    error: str
    code: str
    details: Optional[dict] = None

# HATEOAS links
class UserWithLinks(BaseModel):
    id: int
    name: str
    links: dict = {}
    
    def __init__(self, **data):
        super().__init__(**data)
        self.links = {
            "self": f"/users/{self.id}",
            "orders": f"/users/{self.id}/orders",
            "profile": f"/users/{self.id}/profile"
        }

Query ParamSimple to implementCan be ignoredGradual migration
Content TypeSemantic, RESTfulComplex implementationStrict REST

Best Practices

  1. Use Nouns for Resources: /users not /getUsers
  2. Consistent Naming: Plural nouns, kebab-case for multi-word
  3. Proper Status Codes: 201 for create, 204 for delete, 4xx for client errors
  4. Version from Day One: Plan for breaking changes
  5. Document Everything: Use OpenAPI/Swagger specifications
  6. Support Filtering: Query parameters for search, sort, pagination
