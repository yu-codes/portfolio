Performance optimization through strategic indexing, caching, and query tuning is essential for scalable database systems.

Index Types and Strategies

B-Tree Index (Default)

-- Single column index
CREATE INDEX idx_users_email ON users(email);

-- Composite index (column order matters!)
-- Good for: WHERE status = 'active' AND created_at > '2024-01-01'
-- Good for: WHERE status = 'active'
-- Not optimal for: WHERE created_at > '2024-01-01' alone
CREATE INDEX idx_orders_status_created ON orders(status, created_at);

-- Partial index (index subset of rows)
CREATE INDEX idx_active_users ON users(email) 
WHERE status = 'active';

-- Covering index (includes all query columns)
CREATE INDEX idx_orders_covering ON orders(user_id, status) 
INCLUDE (total_amount, created_at);

Specialized Index Types

-- GIN index for array/JSONB containment
CREATE INDEX idx_products_tags ON products USING GIN (tags);
CREATE INDEX idx_products_attrs ON products USING GIN (attributes jsonb_path_ops);

-- Query using GIN
SELECT * FROM products WHERE tags @> ARRAY['electronics', 'sale'];
SELECT * FROM products WHERE attributes @> '{"color": "red"}';

-- GiST index for geometric/range queries
CREATE INDEX idx_locations_geo ON locations USING GIST (coordinates);

-- BRIN index for large, naturally ordered tables
CREATE INDEX idx_events_created ON events USING BRIN (created_at);

-- Full-text search index
CREATE INDEX idx_articles_search ON articles 
USING GIN (to_tsvector('english', title || ' ' || content));

Query Analysis and Optimization

-- Analyze query execution
EXPLAIN (ANALYZE, BUFFERS, FORMAT TEXT)
SELECT u.name, COUNT(o.id) as order_count
FROM users u
LEFT JOIN orders o ON u.id = o.user_id
WHERE u.created_at > '2024-01-01'
GROUP BY u.id
HAVING COUNT(o.id) > 5
ORDER BY order_count DESC
LIMIT 10;

-- Sample output:
-- Limit  (cost=1234.56..1234.58 rows=10 width=40) (actual time=12.345..12.350 rows=10 loops=1)
--   Buffers: shared hit=500 read=50
--   ->  Sort  (cost=1234.56..1245.67 rows=100 width=40)
--         Sort Key: (count(o.id)) DESC
--         Sort Method: quicksort  Memory: 25kB
--         ->  HashAggregate  (cost=1000.00..1100.00 rows=100 width=40)
--               Filter: (count(o.id) > 5)
--               ->  Hash Left Join  (cost=500.00..900.00 rows=1000 width=36)
--                     Hash Cond: (u.id = o.user_id)
--                     ->  Index Scan using idx_users_created on users u
--                           Index Cond: (created_at > '2024-01-01')

Common Query Optimizations

-- BEFORE: Full table scan
SELECT * FROM orders WHERE YEAR(created_at) = 2024;

-- AFTER: Index-friendly with range
SELECT * FROM orders 
WHERE created_at >= '2024-01-01' AND created_at < '2025-01-01';

-- BEFORE: Function on indexed column
SELECT * FROM users WHERE LOWER(email) = 'user@example.com';

-- AFTER: Expression index
CREATE INDEX idx_users_email_lower ON users(LOWER(email));
SELECT * FROM users WHERE LOWER(email) = 'user@example.com';

-- Or use case-insensitive collation
CREATE INDEX idx_users_email_ci ON users(email COLLATE "en_US.utf8");

Multi-Level Caching

import hashlib
from typing import Optional, Callable, TypeVar
from functools import wraps
import redis.asyncio as redis

T = TypeVar('T')

class CacheService:
    def __init__(self, redis_url: str):
        self.redis = redis.from_url(redis_url)
        self.local_cache: dict = {}  # L1: In-memory
    
    async def get_or_set(
        self,
        key: str,
        factory: Callable[[], T],
        ttl_seconds: int = 300,
        use_local: bool = True
    ) -> T:
        # L1: Check local cache
        if use_local and key in self.local_cache:
            return self.local_cache[key]
        
        # L2: Check Redis
        cached = await self.redis.get(key)
        if cached:
            value = json.loads(cached)
            if use_local:
                self.local_cache[key] = value
            return value
        
        # L3: Compute and cache
        value = await factory() if asyncio.iscoroutinefunction(factory) else factory()
        
        # Write-through to both levels
        await self.redis.setex(key, ttl_seconds, json.dumps(value))
        if use_local:
            self.local_cache[key] = value
        
        return value
    
    async def invalidate(self, pattern: str):
        """Invalidate all keys matching pattern."""
        # Clear local cache
        self.local_cache = {
            k: v for k, v in self.local_cache.items() 
            if not fnmatch.fnmatch(k, pattern)
        }
        # Clear Redis
        async for key in self.redis.scan_iter(match=pattern):
            await self.redis.delete(key)


def cached(key_template: str, ttl: int = 300):
    """Decorator for caching function results."""
    def decorator(func):
        @wraps(func)
        async def wrapper(self, *args, **kwargs):
            # Build cache key from template
            cache_key = key_template.format(*args, **kwargs)
            
            return await self.cache.get_or_set(
                cache_key,
                lambda: func(self, *args, **kwargs),
                ttl_seconds=ttl
            )
        return wrapper
    return decorator


class UserService:
    def __init__(self, db, cache: CacheService):
        self.db = db
        self.cache = cache
    
    @cached("user:{user_id}", ttl=600)
    async def get_user(self, user_id: str) -> dict:
        return await self.db.fetch_user(user_id)
    
    async def update_user(self, user_id: str, data: dict):
        await self.db.update_user(user_id, data)
        await self.cache.invalidate(f"user:{user_id}*")

Best Practices

  1. Index Selectively: Too many indexes slow writes
  2. Use Covering Indexes: Avoid table lookups for common queries
  3. Monitor Index Usage: Drop unused indexes
  4. Cache at Multiple Levels: L1 (process), L2 (Redis), L3 (DB)
  5. Invalidate Proactively: Don’t wait for TTL for critical updates
  6. Analyze Regularly: Run ANALYZE after bulk operations
