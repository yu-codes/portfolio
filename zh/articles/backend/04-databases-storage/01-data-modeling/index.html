Effective data modeling is the foundation of reliable, performant database systems. Proper normalization eliminates redundancy while denormalization optimizes for specific query patterns.

Entity-Relationship Modeling

ER Diagram to Schema

-- Example: E-commerce Domain Model

-- Users entity
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    name VARCHAR(100) NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Products entity
CREATE TABLE products (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    sku VARCHAR(50) UNIQUE NOT NULL,
    name VARCHAR(255) NOT NULL,
    description TEXT,
    price DECIMAL(10, 2) NOT NULL,
    stock_quantity INT DEFAULT 0,
    category_id UUID REFERENCES categories(id),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Orders entity (1:N with users, N:M with products)
CREATE TABLE orders (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id),
    status VARCHAR(20) DEFAULT 'pending',
    total_amount DECIMAL(12, 2) NOT NULL,
    shipping_address JSONB NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Junction table for N:M relationship
CREATE TABLE order_items (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    order_id UUID NOT NULL REFERENCES orders(id) ON DELETE CASCADE,
    product_id UUID NOT NULL REFERENCES products(id),
    quantity INT NOT NULL CHECK (quantity > 0),
    unit_price DECIMAL(10, 2) NOT NULL,
    UNIQUE(order_id, product_id)
);

Normalization Forms

Unnormalized to 3NF Example

-- Unnormalized (UNF): Repeating groups
-- | OrderID | Customer | Items              | Prices       |
-- | 1       | John     | Widget, Gadget     | 10.00, 25.00 |

-- 1NF: Atomic values, no repeating groups
CREATE TABLE orders_1nf (
    order_id INT,
    customer_name VARCHAR(100),
    customer_email VARCHAR(255),
    customer_address TEXT,
    item_name VARCHAR(100),
    item_price DECIMAL(10, 2),
    PRIMARY KEY (order_id, item_name)
);

-- 2NF: Remove partial dependencies
-- (All non-key attributes depend on entire primary key)
CREATE TABLE orders_2nf (
    order_id INT PRIMARY KEY,
    customer_name VARCHAR(100),
    customer_email VARCHAR(255),
    customer_address TEXT
);

CREATE TABLE order_items_2nf (
    order_id INT REFERENCES orders_2nf(order_id),
    item_name VARCHAR(100),
    item_price DECIMAL(10, 2),
    PRIMARY KEY (order_id, item_name)
);

-- 3NF: Remove transitive dependencies
-- (Non-key attributes depend only on primary key)
CREATE TABLE customers (
    customer_id SERIAL PRIMARY KEY,
    name VARCHAR(100),
    email VARCHAR(255) UNIQUE,
    address TEXT
);

CREATE TABLE orders_3nf (
    order_id SERIAL PRIMARY KEY,
    customer_id INT REFERENCES customers(customer_id),
    order_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE items (
    item_id SERIAL PRIMARY KEY,
    name VARCHAR(100),
    price DECIMAL(10, 2)
);

CREATE TABLE order_items_3nf (
    order_id INT REFERENCES orders_3nf(order_id),
    item_id INT REFERENCES items(item_id),
    quantity INT DEFAULT 1,
    PRIMARY KEY (order_id, item_id)
);

Denormalization Strategies

-- Denormalized for read performance
CREATE TABLE order_summary (
    order_id UUID PRIMARY KEY,
    user_id UUID NOT NULL,
    user_name VARCHAR(100),  -- Denormalized from users
    user_email VARCHAR(255), -- Denormalized from users
    total_amount DECIMAL(12, 2),
    item_count INT,
    status VARCHAR(20),
    created_at TIMESTAMP
);

-- Materialized view for reporting
CREATE MATERIALIZED VIEW monthly_sales AS
SELECT 
    DATE_TRUNC('month', o.created_at) AS month,
    p.category_id,
    c.name AS category_name,
    SUM(oi.quantity * oi.unit_price) AS revenue,
    SUM(oi.quantity) AS units_sold
FROM orders o
JOIN order_items oi ON o.id = oi.order_id
JOIN products p ON oi.product_id = p.id
JOIN categories c ON p.category_id = c.id
GROUP BY DATE_TRUNC('month', o.created_at), p.category_id, c.name;

-- Refresh strategy
REFRESH MATERIALIZED VIEW CONCURRENTLY monthly_sales;

Schema Design Patterns

Audit Trail Pattern

-- Audit table for tracking changes
CREATE TABLE audit_log (
    id BIGSERIAL PRIMARY KEY,
    table_name VARCHAR(100) NOT NULL,
    record_id UUID NOT NULL,
    action VARCHAR(10) NOT NULL, -- INSERT, UPDATE, DELETE
    old_data JSONB,
    new_data JSONB,
    changed_by UUID REFERENCES users(id),
    changed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Trigger function
CREATE OR REPLACE FUNCTION audit_trigger()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO audit_log (table_name, record_id, action, old_data, new_data)
    VALUES (
        TG_TABLE_NAME,
        COALESCE(NEW.id, OLD.id),
        TG_OP,
        CASE WHEN TG_OP = 'DELETE' THEN row_to_json(OLD) ELSE NULL END,
        CASE WHEN TG_OP != 'DELETE' THEN row_to_json(NEW) ELSE NULL END
    );
    RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER users_audit
AFTER INSERT OR UPDATE OR DELETE ON users
FOR EACH ROW EXECUTE FUNCTION audit_trigger();

Best Practices

  1. Start Normalized: Begin with 3NF and denormalize based on query patterns
  2. Use Foreign Keys: Enforce referential integrity at the database level
  3. Choose Appropriate Types: Use UUID for distributed systems, SERIAL for simplicity
  4. Document Relationships: Maintain ER diagrams alongside schema
  5. Plan for Growth: Consider partitioning and sharding from the start
  6. Index Thoughtfully: Create indexes based on actual query patterns
