Service Implementation

# Order Service - FastAPI
from fastapi import FastAPI, HTTPException, Depends
from pydantic import BaseModel
from typing import List
import httpx

app = FastAPI(title="Order Service")

class OrderItem(BaseModel):
    product_id: str
    quantity: int

class CreateOrderRequest(BaseModel):
    user_id: str
    items: List[OrderItem]

class OrderService:
    def __init__(self, db, product_client, payment_client):
        self.db = db
        self.product_client = product_client
        self.payment_client = payment_client
    
    async def create_order(self, request: CreateOrderRequest) -> Order:
        # Validate products exist and have stock
        products = await self.product_client.validate_products(
            [item.product_id for item in request.items]
        )
        
        # Calculate total
        total = sum(
            products[item.product_id].price * item.quantity
            for item in request.items
        )
        
        # Create order
        order = Order(
            id=str(uuid.uuid4()),
            user_id=request.user_id,
            items=request.items,
            total=total,
            status="pending"
        )
        await self.db.save(order)
        
        # Reserve inventory (saga pattern)
        try:
            await self.product_client.reserve_inventory(order.id, request.items)
        except Exception as e:
            order.status = "failed"
            await self.db.update(order)
            raise
        
        return order

# Service client with circuit breaker
class ProductServiceClient:
    def __init__(self, base_url: str):
        self.base_url = base_url
        self.client = httpx.AsyncClient(timeout=5.0)
    
    @circuit_breaker(failure_threshold=5, recovery_timeout=30)
    async def validate_products(self, product_ids: List[str]) -> dict:
        response = await self.client.post(
            f"{self.base_url}/products/validate",
            json={"product_ids": product_ids}
        )
        response.raise_for_status()
        return response.json()

Asynchronous: Message Queue

import aio_pika
import json

class EventPublisher:
    def __init__(self, amqp_url: str):
        self.amqp_url = amqp_url
        self.connection = None
        self.channel = None
    
    async def connect(self):
        self.connection = await aio_pika.connect_robust(self.amqp_url)
        self.channel = await self.connection.channel()
    
    async def publish(self, exchange: str, routing_key: str, event: dict):
        exchange_obj = await self.channel.declare_exchange(
            exchange, aio_pika.ExchangeType.TOPIC, durable=True
        )
        
        message = aio_pika.Message(
            body=json.dumps(event).encode(),
            content_type="application/json",
            delivery_mode=aio_pika.DeliveryMode.PERSISTENT
        )
        
        await exchange_obj.publish(message, routing_key=routing_key)

class EventConsumer:
    async def start_consuming(self, queue_name: str, handler):
        connection = await aio_pika.connect_robust(self.amqp_url)
        channel = await connection.channel()
        await channel.set_qos(prefetch_count=10)
        
        queue = await channel.declare_queue(queue_name, durable=True)
        
        async with queue.iterator() as queue_iter:
            async for message in queue_iter:
                async with message.process():
                    event = json.loads(message.body)
                    await handler(event)

Saga Pattern for Distributed Transactions

from enum import Enum
from dataclasses import dataclass
from typing import List, Callable

class SagaStatus(Enum):
    PENDING = "pending"
    COMPLETED = "completed"
    COMPENSATING = "compensating"
    FAILED = "failed"

@dataclass
class SagaStep:
    name: str
    action: Callable
    compensation: Callable

class SagaOrchestrator:
    def __init__(self, saga_id: str, steps: List[SagaStep]):
        self.saga_id = saga_id
        self.steps = steps
        self.completed_steps: List[str] = []
        self.status = SagaStatus.PENDING
    
    async def execute(self, context: dict) -> dict:
        try:
            for step in self.steps:
                result = await step.action(context)
                context.update(result or {})
                self.completed_steps.append(step.name)
            
            self.status = SagaStatus.COMPLETED
            return context
            
        except Exception as e:
            self.status = SagaStatus.COMPENSATING
            await self.compensate(context)
            self.status = SagaStatus.FAILED
            raise SagaFailedError(str(e))
    
    async def compensate(self, context: dict):
        for step_name in reversed(self.completed_steps):
            step = next(s for s in self.steps if s.name == step_name)
            try:
                await step.compensation(context)
            except Exception as e:
                # Log compensation failure, may need manual intervention
                logger.error(f"Compensation failed for {step_name}: {e}")

Best Practices

  1. Design for Failure: Implement circuit breakers, retries, timeouts
  2. Keep Services Small: One bounded context per service
  3. Database per Service: Avoid shared databases
  4. Use Async Communication: Prefer events over synchronous calls
  5. Implement Observability: Distributed tracing, centralized logging
  6. Version Your APIs: Support multiple versions during migration
