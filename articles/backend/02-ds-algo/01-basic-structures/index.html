Array and Dynamic Array

Implementation and Operations

class DynamicArray:
    """Resizable array implementation."""
    
    def __init__(self, capacity: int = 10):
        self._data = [None] * capacity
        self._size = 0
        self._capacity = capacity
    
    def append(self, value) -> None:
        """O(1) amortized - O(n) when resizing."""
        if self._size == self._capacity:
            self._resize(self._capacity * 2)
        self._data[self._size] = value
        self._size += 1
    
    def insert(self, index: int, value) -> None:
        """O(n) - must shift elements."""
        if self._size == self._capacity:
            self._resize(self._capacity * 2)
        for i in range(self._size, index, -1):
            self._data[i] = self._data[i - 1]
        self._data[index] = value
        self._size += 1
    
    def get(self, index: int):
        """O(1) - direct access."""
        if 0 <= index < self._size:
            return self._data[index]
        raise IndexError("Index out of bounds")
    
    def _resize(self, new_capacity: int) -> None:
        new_data = [None] * new_capacity
        for i in range(self._size):
            new_data[i] = self._data[i]
        self._data = new_data
        self._capacity = new_capacity

Linked List

from typing import Optional, Generic, TypeVar

T = TypeVar('T')

class ListNode(Generic[T]):
    def __init__(self, value: T):
        self.value = value
        self.next: Optional[ListNode[T]] = None

class LinkedList(Generic[T]):
    def __init__(self):
        self._head: Optional[ListNode[T]] = None
        self._tail: Optional[ListNode[T]] = None
        self._size = 0
    
    def append(self, value: T) -> None:
        """O(1) - with tail pointer."""
        node = ListNode(value)
        if not self._head:
            self._head = self._tail = node
        else:
            self._tail.next = node
            self._tail = node
        self._size += 1
    
    def prepend(self, value: T) -> None:
        """O(1) - insert at head."""
        node = ListNode(value)
        node.next = self._head
        self._head = node
        if not self._tail:
            self._tail = node
        self._size += 1
    
    def find(self, value: T) -> Optional[ListNode[T]]:
        """O(n) - linear search."""
        current = self._head
        while current:
            if current.value == value:
                return current
            current = current.next
        return None

Hash Map

class HashMap:
    """Hash map with separate chaining."""
    
    def __init__(self, capacity: int = 16, load_factor: float = 0.75):
        self._buckets: list[list[tuple]] = [[] for _ in range(capacity)]
        self._size = 0
        self._load_factor = load_factor
    
    def put(self, key, value) -> None:
        """O(1) average, O(n) worst case."""
        if self._size / len(self._buckets) > self._load_factor:
            self._resize()
        
        bucket_idx = hash(key) % len(self._buckets)
        bucket = self._buckets[bucket_idx]
        
        for i, (k, v) in enumerate(bucket):
            if k == key:
                bucket[i] = (key, value)
                return
        
        bucket.append((key, value))
        self._size += 1
    
    def get(self, key, default=None):
        """O(1) average."""
        bucket_idx = hash(key) % len(self._buckets)
        for k, v in self._buckets[bucket_idx]:
            if k == key:
                return v
        return default
    
    def _resize(self) -> None:
        old_buckets = self._buckets
        self._buckets = [[] for _ in range(len(old_buckets) * 2)]
        self._size = 0
        for bucket in old_buckets:
            for key, value in bucket:
                self.put(key, value)

Stack and Queue

from collections import deque

class Stack(Generic[T]):
    """LIFO stack using list."""
    
    def __init__(self):
        self._items: list[T] = []
    
    def push(self, item: T) -> None:
        """O(1) amortized."""
        self._items.append(item)
    
    def pop(self) -> T:
        """O(1)."""
        if self.is_empty():
            raise IndexError("Stack is empty")
        return self._items.pop()
    
    def peek(self) -> T:
        """O(1)."""
        if self.is_empty():
            raise IndexError("Stack is empty")
        return self._items[-1]
    
    def is_empty(self) -> bool:
        return len(self._items) == 0


class Queue(Generic[T]):
    """FIFO queue using deque for O(1) operations."""
    
    def __init__(self):
        self._items: deque[T] = deque()
    
    def enqueue(self, item: T) -> None:
        """O(1)."""
        self._items.append(item)
    
    def dequeue(self) -> T:
        """O(1)."""
        if self.is_empty():
            raise IndexError("Queue is empty")
        return self._items.popleft()
    
    def peek(self) -> T:
        """O(1)."""
        if self.is_empty():
            raise IndexError("Queue is empty")
        return self._items[0]

Binary Tree

class TreeNode(Generic[T]):
    def __init__(self, value: T):
        self.value = value
        self.left: Optional[TreeNode[T]] = None
        self.right: Optional[TreeNode[T]] = None

class BinarySearchTree(Generic[T]):
    def __init__(self):
        self._root: Optional[TreeNode[T]] = None
    
    def insert(self, value: T) -> None:
        """O(log n) average, O(n) worst case."""
        self._root = self._insert_recursive(self._root, value)
    
    def _insert_recursive(self, node: Optional[TreeNode[T]], value: T) -> TreeNode[T]:
        if not node:
            return TreeNode(value)
        if value < node.value:
            node.left = self._insert_recursive(node.left, value)
        else:
            node.right = self._insert_recursive(node.right, value)
        return node
    
    def search(self, value: T) -> Optional[TreeNode[T]]:
        """O(log n) average."""
        current = self._root
        while current:
            if value == current.value:
                return current
            elif value < current.value:
                current = current.left
            else:
                current = current.right
        return None

Best Practices

  1. Use Built-in Types: Python list, dict, deque are highly optimized
  2. Consider Access Patterns: Random access → Array, Sequential → LinkedList
  3. Profile Before Optimizing: Measure actual performance bottlenecks
  4. Understand Trade-offs: Memory vs speed, simplicity vs efficiency
  5. Use Type Hints: Improve code clarity and catch errors early
