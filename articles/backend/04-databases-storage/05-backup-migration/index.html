Backup Strategies

PostgreSQL Backup Methods

# Logical backup with pg_dump
pg_dump -h localhost -U postgres -d mydb \
  --format=custom \
  --compress=9 \
  --file=backup_$(date +%Y%m%d_%H%M%S).dump

# Parallel dump for large databases
pg_dump -h localhost -U postgres -d mydb \
  --format=directory \
  --jobs=4 \
  --file=backup_dir/

# Dump specific tables
pg_dump -h localhost -U postgres -d mydb \
  --table=users --table=orders \
  --data-only \
  --file=partial_backup.sql

# Restore from custom format
pg_restore -h localhost -U postgres -d mydb_new \
  --jobs=4 \
  --verbose \
  backup.dump

Automated Backup with Python

import subprocess
import boto3
from datetime import datetime, timedelta
from pathlib import Path
import logging

class BackupManager:
    def __init__(self, config: dict):
        self.config = config
        self.s3 = boto3.client('s3')
        self.logger = logging.getLogger(__name__)
    
    async def create_backup(self) -> str:
        timestamp = datetime.utcnow().strftime('%Y%m%d_%H%M%S')
        backup_file = f"backup_{timestamp}.dump"
        backup_path = Path(self.config['backup_dir']) / backup_file
        
        # Create dump
        cmd = [
            'pg_dump',
            f"--host={self.config['db_host']}",
            f"--username={self.config['db_user']}",
            f"--dbname={self.config['db_name']}",
            '--format=custom',
            '--compress=9',
            f"--file={backup_path}"
        ]
        
        env = {'PGPASSWORD': self.config['db_password']}
        result = subprocess.run(cmd, env=env, capture_output=True)
        
        if result.returncode != 0:
            raise BackupError(f"Backup failed: {result.stderr.decode()}")
        
        # Upload to S3
        s3_key = f"backups/{backup_file}"
        self.s3.upload_file(str(backup_path), self.config['s3_bucket'], s3_key)
        
        # Cleanup local file
        backup_path.unlink()
        
        self.logger.info(f"Backup completed: {s3_key}")
        return s3_key
    
    async def cleanup_old_backups(self, retention_days: int = 30):
        cutoff = datetime.utcnow() - timedelta(days=retention_days)
        
        paginator = self.s3.get_paginator('list_objects_v2')
        for page in paginator.paginate(Bucket=self.config['s3_bucket'], Prefix='backups/'):
            for obj in page.get('Contents', []):
                if obj['LastModified'].replace(tzinfo=None) < cutoff:
                    self.s3.delete_object(
                        Bucket=self.config['s3_bucket'],
                        Key=obj['Key']
                    )
                    self.logger.info(f"Deleted old backup: {obj['Key']}")

Schema Migration with Alembic

# alembic/versions/001_create_users.py
"""Create users table

Revision ID: 001
Create Date: 2024-01-15 10:00:00
"""
from alembic import op
import sqlalchemy as sa

revision = '001'
down_revision = None

def upgrade():
    op.create_table(
        'users',
        sa.Column('id', sa.UUID(), primary_key=True),
        sa.Column('email', sa.String(255), nullable=False, unique=True),
        sa.Column('name', sa.String(100), nullable=False),
        sa.Column('created_at', sa.DateTime(), server_default=sa.func.now()),
    )
    op.create_index('idx_users_email', 'users', ['email'])

def downgrade():
    op.drop_index('idx_users_email')
    op.drop_table('users')

# alembic/versions/002_add_user_status.py
"""Add status column to users

Revision ID: 002
Create Date: 2024-01-20 10:00:00
"""
from alembic import op
import sqlalchemy as sa

revision = '002'
down_revision = '001'

def upgrade():
    # Add column with default
    op.add_column(
        'users',
        sa.Column('status', sa.String(20), server_default='active')
    )
    # Backfill existing rows
    op.execute("UPDATE users SET status = 'active' WHERE status IS NULL")
    # Make non-nullable
    op.alter_column('users', 'status', nullable=False)

def downgrade():
    op.drop_column('users', 'status')

Zero-Downtime Migration Strategies

def upgrade_phase1():

def upgrade_phase3():
    op.create_unique_constraint('uq_users_new_email', 'users', ['new_email'])
    op.alter_column('users', 'new_email', nullable=False)

# Phase 4: Remove old column (after app is fully migrated)
def upgrade_phase4():
    op.drop_column('users', 'email')
    op.alter_column('users', 'new_email', new_column_name='email')

Recovery Procedures

# Test restore to temporary database
createdb -h localhost -U postgres mydb_test

# Verify data integrity
psql -h localhost -U postgres -d mydb_test -c "
SELECT 
  (SELECT COUNT(*) FROM users) as user_count,
  (SELECT COUNT(*) FROM orders) as order_count;
"

# Point-in-time recovery
mkdir /var/lib/postgresql/recovery
tar -xzf base_backup.tar.gz -C /var/lib/postgresql/recovery

cat > /var/lib/postgresql/recovery/postgresql.auto.conf << EOF
restore_command = 'cp /archive/%f %p'
recovery_target_time = '2024-01-15 14:30:00 UTC'
recovery_target_action = 'promote'
EOF

touch /var/lib/postgresql/recovery/recovery.signal
pg_ctl start -D /var/lib/postgresql/recovery

Best Practices

  1. Test Restores Regularly: Verify backups actually work
  2. Automate Everything: Schedule backups and retention cleanup
  3. Use Multiple Locations: Store backups in different regions
  4. Monitor Backup Health: Alert on backup failures
  5. Document Procedures: Maintain runbooks for recovery
  6. Practice Migrations: Test migrations in staging before production
