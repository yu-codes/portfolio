Window Functions and CTEs

-- Window functions for analytics
SELECT 
    user_id,
    order_date,
    total_amount,
    -- Running total
    SUM(total_amount) OVER (
        PARTITION BY user_id 
        ORDER BY order_date
    ) AS cumulative_total,
    -- Rank within user's orders
    ROW_NUMBER() OVER (
        PARTITION BY user_id 
        ORDER BY total_amount DESC
    ) AS amount_rank,
    -- Previous order amount
    LAG(total_amount) OVER (
        PARTITION BY user_id 
        ORDER BY order_date
    ) AS prev_order_amount
FROM orders;

-- Recursive CTE for hierarchical data
WITH RECURSIVE category_tree AS (
    -- Base case: root categories
    SELECT id, name, parent_id, 1 AS depth, name::TEXT AS path
    FROM categories
    WHERE parent_id IS NULL
    
    UNION ALL
    
    -- Recursive case
    SELECT c.id, c.name, c.parent_id, ct.depth + 1,
           ct.path || ' > ' || c.name
    FROM categories c
    INNER JOIN category_tree ct ON c.parent_id = ct.id
)
SELECT * FROM category_tree ORDER BY path;

JSONB Operations

-- JSONB column for flexible data
CREATE TABLE products (
    id UUID PRIMARY KEY,
    name VARCHAR(255),
    attributes JSONB DEFAULT '{}',
    CONSTRAINT valid_json CHECK (jsonb_typeof(attributes) = 'object')
);

-- Query JSON data
SELECT * FROM products 
WHERE attributes @> '{"color": "red"}';

SELECT * FROM products 
WHERE attributes->>'brand' = 'Apple';

SELECT * FROM products 
WHERE (attributes->>'price')::numeric > 100;

-- JSON aggregation
SELECT 
    category_id,
    jsonb_agg(jsonb_build_object(
        'id', id,
        'name', name,
        'price', attributes->>'price'
    )) AS products
FROM products
GROUP BY category_id;

-- GIN index for JSON queries
CREATE INDEX idx_products_attributes ON products USING GIN (attributes);

Partitioning

-- Range partitioning by date
CREATE TABLE events (
    id UUID DEFAULT gen_random_uuid(),
    event_type VARCHAR(50),
    payload JSONB,
    created_at TIMESTAMP NOT NULL
) PARTITION BY RANGE (created_at);

-- Create partitions
CREATE TABLE events_2024_q1 PARTITION OF events
    FOR VALUES FROM ('2024-01-01') TO ('2024-04-01');

CREATE TABLE events_2024_q2 PARTITION OF events
    FOR VALUES FROM ('2024-04-01') TO ('2024-07-01');

-- List partitioning
CREATE TABLE orders (
    id UUID,
    region VARCHAR(50),
    total DECIMAL(10,2)
) PARTITION BY LIST (region);

CREATE TABLE orders_north PARTITION OF orders
    FOR VALUES IN ('US', 'CA');

CREATE TABLE orders_asia PARTITION OF orders
    FOR VALUES IN ('JP', 'KR', 'TW');

Transaction Isolation Levels

-- Read Committed (PostgreSQL default)
BEGIN;
SET TRANSACTION ISOLATION LEVEL READ COMMITTED;
-- Sees only committed data, may see different 
-- snapshots within transaction
COMMIT;

-- Repeatable Read
BEGIN;
SET TRANSACTION ISOLATION LEVEL REPEATABLE READ;
-- Sees consistent snapshot from start of transaction
-- Prevents non-repeatable reads
COMMIT;

-- Serializable (strictest)
BEGIN;
SET TRANSACTION ISOLATION LEVEL SERIALIZABLE;
-- Full ACID isolation, as if transactions were serial
-- May cause serialization failures - must retry
COMMIT;

# Python with retry logic for serializable
import psycopg
from tenacity import retry, retry_if_exception_type

class SerializationError(Exception):
    pass

@retry(retry=retry_if_exception_type(SerializationError), stop=3)
async def transfer_funds(from_id: str, to_id: str, amount: Decimal):
    async with pool.connection() as conn:
        try:
            async with conn.transaction():
                await conn.execute(
                    "SET TRANSACTION ISOLATION LEVEL SERIALIZABLE"
                )
                
                # Deduct from source
                result = await conn.execute(
                    "UPDATE accounts SET balance = balance - %s "
                    "WHERE id = %s AND balance >= %s RETURNING balance",
                    [amount, from_id, amount]
                )
                if not result.rowcount:
                    raise InsufficientFunds()
                
                # Add to destination
                await conn.execute(
                    "UPDATE accounts SET balance = balance + %s WHERE id = %s",
                    [amount, to_id]
                )
        except psycopg.errors.SerializationFailure:
            raise SerializationError("Retry needed")

Connection Pooling

# Python with connection pooling
import asyncpg
from contextlib import asynccontextmanager

class DatabasePool:
    def __init__(self, dsn: str, min_size: int = 10, max_size: int = 50):
        self.dsn = dsn
        self.min_size = min_size
        self.max_size = max_size
        self._pool = None
    
    async def connect(self):
        self._pool = await asyncpg.create_pool(
            dsn=self.dsn,
            min_size=self.min_size,
            max_size=self.max_size,
            command_timeout=60,
            statement_cache_size=100
        )
    
    async def close(self):
        if self._pool:
            await self._pool.close()
    
    @asynccontextmanager
    async def acquire(self):
        async with self._pool.acquire() as conn:
            yield conn
    
    async def execute(self, query: str, *args):
        async with self.acquire() as conn:
            return await conn.execute(query, *args)
    
    async def fetch(self, query: str, *args):
        async with self.acquire() as conn:
            return await conn.fetch(query, *args)

Best Practices

  1. Use Connection Pooling: Reuse connections to reduce overhead
  2. Index Based on Queries: Analyze query patterns with EXPLAIN ANALYZE
  3. Use Appropriate Isolation: Match isolation level to use case
  4. Partition Large Tables: Improve query performance and maintenance
  5. Monitor Performance: Use pg_stat_statements, slow query logs
  6. Regular Maintenance: VACUUM, ANALYZE, index maintenance
