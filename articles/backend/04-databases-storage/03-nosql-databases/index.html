Schema Design and Operations

from motor.motor_asyncio import AsyncIOMotorClient
from pydantic import BaseModel, Field
from datetime import datetime
from typing import List, Optional
import pymongo

class Address(BaseModel):
    street: str
    city: str
    country: str
    zip_code: str

class Order(BaseModel):
    product_id: str
    quantity: int
    price: float
    ordered_at: datetime = Field(default_factory=datetime.utcnow)

class User(BaseModel):
    id: str = Field(alias="_id")
    email: str
    name: str
    addresses: List[Address] = []
    recent_orders: List[Order] = []  # Embedded documents
    created_at: datetime = Field(default_factory=datetime.utcnow)

class MongoRepository:
    def __init__(self, uri: str, database: str):
        self.client = AsyncIOMotorClient(uri)
        self.db = self.client[database]
        self.users = self.db.users
    
    async def setup_indexes(self):
        # Compound index
        await self.users.create_index([
            ("email", pymongo.ASCENDING)
        ], unique=True)
        
        # Text index for search
        await self.users.create_index([
            ("name", pymongo.TEXT),
            ("email", pymongo.TEXT)
        ])
        
        # TTL index for expiration
        await self.db.sessions.create_index(
            "expires_at",
            expireAfterSeconds=0
        )
    
    async def find_users_with_orders(self, min_amount: float):
        # Aggregation pipeline
        pipeline = [
            {"$unwind": "$recent_orders"},
            {"$group": {
                "_id": "$_id",
                "email": {"$first": "$email"},
                "total_spent": {"$sum": {
                    "$multiply": [
                        "$recent_orders.price",
                        "$recent_orders.quantity"
                    ]
                }}
            }},
            {"$match": {"total_spent": {"$gte": min_amount}}},
            {"$sort": {"total_spent": -1}}
        ]
        return await self.users.aggregate(pipeline).to_list(100)
    
    async def add_order(self, user_id: str, order: Order):
        # Atomic update with array push
        await self.users.update_one(
            {"_id": user_id},
            {
                "$push": {
                    "recent_orders": {
                        "$each": [order.dict()],
                        "$slice": -10  # Keep only last 10 orders
                    }
                }
            }
        )

Redis In-Memory Store

import redis.asyncio as redis
import json
from typing import Optional, Any
from datetime import timedelta

class RedisCache:
    def __init__(self, url: str = "redis://localhost:6379"):
        self.redis = redis.from_url(url, decode_responses=True)
    
    # String operations
    async def set(
        self,
        key: str,
        value: Any,
        ttl: Optional[timedelta] = None
    ):
        serialized = json.dumps(value)
        await self.redis.set(key, serialized, ex=ttl)
    
    async def get(self, key: str) -> Optional[Any]:
        data = await self.redis.get(key)
        return json.loads(data) if data else None
    
    # Hash operations for objects
    async def hset(self, key: str, mapping: dict):
        await self.redis.hset(key, mapping=mapping)
    
    async def hget(self, key: str, field: str) -> Optional[str]:
        return await self.redis.hget(key, field)
    
    # Sorted sets for leaderboards
    async def add_score(self, leaderboard: str, user_id: str, score: float):
        await self.redis.zadd(leaderboard, {user_id: score})
    
    async def get_top_n(self, leaderboard: str, n: int = 10):
        return await self.redis.zrevrange(
            leaderboard, 0, n - 1, withscores=True
        )
    
    # Rate limiting with sliding window
    async def is_rate_limited(
        self,
        key: str,
        max_requests: int,
        window_seconds: int
    ) -> bool:
        pipe = self.redis.pipeline()
        now = time.time()
        window_start = now - window_seconds
        
        # Remove old entries
        pipe.zremrangebyscore(key, 0, window_start)
        # Add current request
        pipe.zadd(key, {str(now): now})
        # Count requests in window
        pipe.zcard(key)
        # Set expiry
        pipe.expire(key, window_seconds)
        
        results = await pipe.execute()
        return results[2] > max_requests
    
    # Pub/Sub
    async def publish(self, channel: str, message: dict):
        await self.redis.publish(channel, json.dumps(message))
    
    async def subscribe(self, channel: str):
        pubsub = self.redis.pubsub()
        await pubsub.subscribe(channel)
        async for message in pubsub.listen():
            if message["type"] == "message":
                yield json.loads(message["data"])

DynamoDB Single-Table Design

import boto3
from boto3.dynamodb.conditions import Key, Attr
from typing import List, Optional

class DynamoDBRepository:
    def __init__(self, table_name: str):
        self.dynamodb = boto3.resource('dynamodb')
        self.table = self.dynamodb.Table(table_name)
    
    # Single-table design patterns
    async def get_user(self, user_id: str) -> dict:
        response = self.table.get_item(
            Key={
                'PK': f'USER#{user_id}',
                'SK': f'PROFILE#{user_id}'
            }
        )
        return response.get('Item')
    
    async def get_user_orders(self, user_id: str) -> List[dict]:
        response = self.table.query(
            KeyConditionExpression=
                Key('PK').eq(f'USER#{user_id}') &
                Key('SK').begins_with('ORDER#')
        )
        return response.get('Items', [])
    
    async def create_order(self, user_id: str, order: dict):
        # Transaction for consistency
        self.dynamodb.meta.client.transact_write_items(
            TransactItems=[
                {
                    'Put': {
                        'TableName': self.table.name,
                        'Item': {
                            'PK': f'USER#{user_id}',
                            'SK': f'ORDER#{order["id"]}',
                            'GSI1PK': f'ORDER#{order["id"]}',
                            'GSI1SK': f'USER#{user_id}',
                            **order
                        }
                    }
                },
                {
                    'Update': {
                        'TableName': self.table.name,
                        'Key': {
                            'PK': f'USER#{user_id}',
                            'SK': f'PROFILE#{user_id}'
                        },
                        'UpdateExpression': 'ADD order_count :inc',
                        'ExpressionAttributeValues': {':inc': 1}
                    }
                }
            ]
        )

Best Practices

  1. Design for Access Patterns: Model data based on how it will be queried
  2. Embed vs Reference: Embed for 1:1, reference for 1:N with high cardinality
  3. Use Appropriate Indexes: Balance read performance with write overhead
  4. Implement TTL: Auto-expire temporary data to manage storage
  5. Plan for Scale: Design partition keys for even distribution
  6. Monitor Operations: Track slow queries, connection counts, memory usage
