Optimization techniques transform slow, resource-intensive operations into efficient, scalable solutions. This guide covers practical optimization strategies across the application stack.

Introduction

Optimization should be driven by data, not intuition. Always profile first to identify actual bottlenecks, then apply targeted optimizations. Premature optimization is the root of all evil, but ignoring performance in critical paths leads to poor user experience.

Query Optimization Patterns

from sqlalchemy import select, func
from sqlalchemy.orm import joinedload, selectinload, load_only

class OptimizedQueries:
    """Optimized database query patterns."""
    
    def __init__(self, session):
        self.session = session
    
    def get_users_with_orders_n_plus_one(self, limit: int = 100):
        """BAD: N+1 query problem."""
        users = self.session.query(User).limit(limit).all()
        for user in users:
            # This triggers a separate query for each user!
            print(f"{user.name}: {len(user.orders)} orders")
    
    def get_users_with_orders_optimized(self, limit: int = 100):
        """GOOD: Eager loading to avoid N+1."""
        users = (
            self.session.query(User)
            .options(selectinload(User.orders))  # Load orders in single query
            .limit(limit)
            .all()
        )
        return users
    
    def get_users_with_order_count(self, limit: int = 100):
        """BETTER: Aggregate in database."""
        result = (
            self.session.query(
                User.id,
                User.name,
                func.count(Order.id).label('order_count')
            )
            .outerjoin(Order)
            .group_by(User.id, User.name)
            .order_by(func.count(Order.id).desc())
            .limit(limit)
            .all()
        )
        return result
    
    def get_users_minimal_columns(self, user_ids: list[int]):
        """Load only required columns."""
        return (
            self.session.query(User)
            .options(load_only(User.id, User.name, User.email))
            .filter(User.id.in_(user_ids))
            .all()
        )
    
    def bulk_update_optimized(self, user_ids: list[int], status: str):
        """Bulk update instead of individual updates."""
        # BAD: Individual updates
        # for user_id in user_ids:
        #     user = self.session.query(User).get(user_id)
        #     user.status = status
        
        # GOOD: Single bulk update
        self.session.query(User).filter(
            User.id.in_(user_ids)
        ).update(
            {User.status: status},
            synchronize_session='fetch'
        )
        self.session.commit()

Batching Operations

import asyncio
from typing import TypeVar, Callable, Awaitable
from collections import defaultdict
import time

T = TypeVar('T')
R = TypeVar('R')

class BatchProcessor:
    """Batch multiple requests into single operations."""
    
    def __init__(
        self,
        batch_fn: Callable[[list[T]], Awaitable[dict[T, R]]],
        max_batch_size: int = 100,
        max_wait_ms: int = 10
    ):
        self.batch_fn = batch_fn
        self.max_batch_size = max_batch_size
        self.max_wait_ms = max_wait_ms
        self.pending: dict[T, asyncio.Future] = {}
        self.batch: list[T] = []
        self.timer: asyncio.TimerHandle | None = None
        self.lock = asyncio.Lock()
    
    async def load(self, key: T) -> R:
        """Load a single item, batched with other requests."""
        async with self.lock:
            # Check if already pending
            if key in self.pending:
                return await self.pending[key]
            
            # Create future for this key
            future = asyncio.get_event_loop().create_future()
            self.pending[key] = future
            self.batch.append(key)
            
            # Schedule batch execution
            if len(self.batch) >= self.max_batch_size:
                await self._execute_batch()
            elif self.timer is None:
                self.timer = asyncio.get_event_loop().call_later(
                    self.max_wait_ms / 1000,
                    lambda: asyncio.create_task(self._execute_batch())
                )
        
        return await future
    
    async def _execute_batch(self):
        """Execute the batched request."""
        async with self.lock:
            if self.timer:
                self.timer.cancel()
                self.timer = None
            
            if not self.batch:
                return
            
            batch = self.batch
            pending = self.pending
            self.batch = []
            self.pending = {}
        
        try:
            results = await self.batch_fn(batch)
            for key in batch:
                if key in pending:
                    pending[key].set_result(results.get(key))
        except Exception as e:
            for key in batch:
                if key in pending and not pending[key].done():
                    pending[key].set_exception(e)


# DataLoader pattern for GraphQL
class DataLoader:
    """Generic data loader with batching and caching."""
    
    def __init__(self, batch_load_fn: Callable):
        self.batch_load_fn = batch_load_fn
        self.cache: dict = {}
        self.batch_processor = BatchProcessor(self._batch_load)
    
    async def _batch_load(self, keys: list) -> dict:
        """Load batch of keys."""
        # Filter out cached keys
        uncached_keys = [k for k in keys if k not in self.cache]
        
        if uncached_keys:
            results = await self.batch_load_fn(uncached_keys)
            self.cache.update(results)
        
        return {k: self.cache.get(k) for k in keys}
    
    async def load(self, key) -> any:
        """Load single key."""
        if key in self.cache:
            return self.cache[key]
        return await self.batch_processor.load(key)
    
    async def load_many(self, keys: list) -> list:
        """Load multiple keys."""
        return await asyncio.gather(*[self.load(k) for k in keys])
    
    def clear(self, key=None):
        """Clear cache."""
        if key:
            self.cache.pop(key, None)
        else:
            self.cache.clear()

Async Processing

import asyncio
from concurrent.futures import ThreadPoolExecutor
from typing import List, Any
import aiohttp
import aioboto3

class AsyncOptimizations:
    """Async patterns for I/O optimization."""
    
    def __init__(self):
        self.thread_pool = ThreadPoolExecutor(max_workers=10)
        self.semaphore = asyncio.Semaphore(100)  # Limit concurrent requests
    
    async def fetch_urls_sequential(self, urls: List[str]) -> List[dict]:
        """BAD: Sequential fetching."""
        results = []
        async with aiohttp.ClientSession() as session:
            for url in urls:
                async with session.get(url) as response:
                    results.append(await response.json())
        return results
    
    async def fetch_urls_parallel(self, urls: List[str]) -> List[dict]:
        """GOOD: Parallel fetching with semaphore."""
        async def fetch_one(session: aiohttp.ClientSession, url: str):
            async with self.semaphore:
                async with session.get(url) as response:
                    return await response.json()
        
        async with aiohttp.ClientSession() as session:
            tasks = [fetch_one(session, url) for url in urls]
            return await asyncio.gather(*tasks, return_exceptions=True)
    
    async def run_sync_in_thread(self, sync_func, *args):
        """Run blocking sync function in thread pool."""
        loop = asyncio.get_event_loop()
        return await loop.run_in_executor(
            self.thread_pool,
            sync_func,
            *args
        )
    
    async def parallel_database_and_cache(
        self,
        user_id: int,
        cache: Any,
        db: Any
    ):
        """Check cache and database in parallel."""
        cache_task = asyncio.create_task(cache.get(f"user:{user_id}"))
        db_task = asyncio.create_task(db.get_user(user_id))
        
        # Wait for cache first
        cached = await cache_task
        if cached:
            db_task.cancel()
            return cached
        
        # Fall back to database
        user = await db_task
        if user:
            await cache.set(f"user:{user_id}", user)
        return user

Connection Pooling

import asyncpg
import aiohttp
from contextlib import asynccontextmanager

class ConnectionPoolManager:
    """Manage connection pools for various services."""
    
    def __init__(self):
        self.db_pool: asyncpg.Pool | None = None
        self.http_session: aiohttp.ClientSession | None = None
    
    async def init_pools(self):
        """Initialize connection pools."""
        # Database connection pool
        self.db_pool = await asyncpg.create_pool(
            dsn="postgresql://user:pass@localhost/db",
            min_size=10,
            max_size=100,
            max_inactive_connection_lifetime=300,
            command_timeout=30
        )
        
        # HTTP connection pool
        connector = aiohttp.TCPConnector(
            limit=100,           # Total connection limit
            limit_per_host=30,   # Per-host limit
            ttl_dns_cache=300,   # DNS cache TTL
            keepalive_timeout=30
        )
        
        timeout = aiohttp.ClientTimeout(
            total=30,
            connect=5,
            sock_read=10
        )
        
        self.http_session = aiohttp.ClientSession(
            connector=connector,
            timeout=timeout
        )
    
    async def close_pools(self):
        """Close all connection pools."""
        if self.db_pool:
            await self.db_pool.close()
        if self.http_session:
            await self.http_session.close()
    
    @asynccontextmanager
    async def get_db_connection(self):
        """Get database connection from pool."""
        async with self.db_pool.acquire() as conn:
            yield conn
    
    async def execute_query(self, query: str, *args):
        """Execute query using pooled connection."""
        async with self.get_db_connection() as conn:
            return await conn.fetch(query, *args)

Best Practices

  1. Profile before optimizing to identify actual bottlenecks
  2. Measure impact of each optimization with before/after benchmarks
  3. Optimize the critical path first for maximum user impact
  4. Use appropriate data structures for your access patterns
  5. Batch similar operations to reduce overhead
  6. Implement timeouts for all external calls
  7. Monitor performance continuously to catch regressions
  8. Document optimizations so future developers understand trade-offs
