Memory Layout Fundamentals

Stack vs Heap Allocation

package main

// Stack allocation - fixed size, fast, automatic cleanup
func stackExample() int {
    x := 42        // Allocated on stack
    y := x * 2     // Allocated on stack
    return y       // Stack automatically cleaned up
}

// Heap allocation - dynamic size, slower, requires GC
func heapExample() *User {
    user := &User{  // Escapes to heap due to pointer return
        Name: "Alice",
        Data: make([]byte, 1024), // Slice backing array on heap
    }
    return user
}

// Escape analysis determines allocation location
//go:noinline
func escapeAnalysis() {
    // Run: go build -gcflags="-m" to see escape analysis
    local := User{Name: "Bob"}    // Stack (doesn't escape)
    _ = &local                      // Still stack (reference doesn't escape)
    
    global = &User{Name: "Carol"}  // Heap (escapes to global)
}

Python Memory Model

import sys
import gc

# Everything is an object on the heap
def python_memory_demo():
    # Small integers are interned (-5 to 256)
    a = 100
    b = 100
    print(a is b)  # True - same object
    
    # Larger integers create new objects
    x = 1000
    y = 1000
    print(x is y)  # False - different objects
    
    # String interning
    s1 = "hello"
    s2 = "hello"
    print(s1 is s2)  # True - interned
    
    # Check reference counts
    obj = [1, 2, 3]
    print(sys.getrefcount(obj))  # 2 (obj + getrefcount arg)

# Reference counting + cycle detection
class Node:
    def __init__(self):
        self.ref = None

def circular_reference():
    a = Node()
    b = Node()
    a.ref = b
    b.ref = a  # Circular reference!
    
    del a, b
    # Reference count > 0, but unreachable
    # Cycle detector will clean up
    
    gc.collect()  # Force garbage collection
    print(gc.get_stats())  # GC statistics

Garbage Collection Strategies

Java G1 Garbage Collector

// JVM memory regions
// -Xms512m -Xmx2g -XX:+UseG1GC -XX:MaxGCPauseMillis=200

public class MemoryDemo {
    // Young Generation: Eden + Survivor spaces
    // Old Generation: Long-lived objects
    // Metaspace: Class metadata
    
    public static void main(String[] args) {
        // Object lifecycle
        Object shortLived = new byte[1024];  // Eden
        // After Minor GC, if survived -> Survivor
        // After multiple GCs -> Old Generation
        
        // Monitor memory
        Runtime runtime = Runtime.getRuntime();
        long usedMemory = runtime.totalMemory() - runtime.freeMemory();
        System.out.println("Used: " + usedMemory / 1024 / 1024 + " MB");
        
        // Force GC (hint, not guaranteed)
        System.gc();
    }
}

Go Concurrent GC

import (
    "runtime"
    "runtime/debug"
)

func gcTuning() {
    // Set GC target percentage (default 100)
    debug.SetGCPercent(50) // More aggressive
    
    // Set memory limit (Go 1.19+)
    debug.SetMemoryLimit(1 << 30) // 1GB
    
    // Get GC stats
    var stats debug.GCStats
    debug.ReadGCStats(&stats)
    
    // Memory statistics
    var memStats runtime.MemStats
    runtime.ReadMemStats(&memStats)
    
    println("Heap Alloc:", memStats.HeapAlloc)
    println("Num GC:", memStats.NumGC)
}

Execution Models

Python:          Source → Bytecode → CPython Interpreter
Go:              Source → Machine Code (static binary)
Java:            Source → Bytecode → JIT → Machine Code
TypeScript:      Source → JavaScript → V8 JIT → Machine Code

Best Practices

  1. Minimize Allocations: Reuse objects, use object pools for hot paths
  2. Avoid Memory Leaks: Close resources, clear references, use weak references
  3. Profile Memory Usage: Use language-specific profilers (pprof, JVisualVM)
  4. Tune GC for Workload: Configure GC based on latency vs throughput needs
  5. Understand Escape Analysis: Keep objects on stack when possible
