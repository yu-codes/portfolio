A01: Broken Access Control

from functools import wraps
from fastapi import HTTPException, Depends, status
from typing import List

class AccessControl:
    """Implement proper access control checks."""
    
    def __init__(self):
        self.permissions = {
            'admin': ['read', 'write', 'delete', 'admin'],
            'editor': ['read', 'write'],
            'viewer': ['read']
        }
    
    def require_permission(self, permission: str):
        """Decorator to check user permissions."""
        def decorator(func):
            @wraps(func)
            async def wrapper(*args, current_user=None, **kwargs):
                if not current_user:
                    raise HTTPException(
                        status_code=status.HTTP_401_UNAUTHORIZED,
                        detail="Authentication required"
                    )
                
                user_role = current_user.role
                allowed_permissions = self.permissions.get(user_role, [])
                
                if permission not in allowed_permissions:
                    raise HTTPException(
                        status_code=status.HTTP_403_FORBIDDEN,
                        detail="Insufficient permissions"
                    )
                
                return await func(*args, current_user=current_user, **kwargs)
            return wrapper
        return decorator
    
    def check_resource_ownership(self, user_id: int, resource_owner_id: int):
        """Prevent IDOR by checking resource ownership."""
        if user_id != resource_owner_id:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail="Access denied to this resource"
            )


access_control = AccessControl()

# BAD: No access control
@app.get("/users/{user_id}")

    return db.get_user(user_id)  # Any user can access any profile!

# GOOD: Proper access control
@app.get("/users/{user_id}")
@access_control.require_permission('read')
async def get_user_secure(
    user_id: int, 
    current_user: User = Depends(get_current_user)
):
    # Check if user is accessing their own data or is admin
    if current_user.id != user_id and current_user.role != 'admin':
        raise HTTPException(status_code=403, detail="Access denied")
    return db.get_user(user_id)

A02: Cryptographic Failures

from cryptography.fernet import Fernet
import hashlib
import hmac
import secrets

class SecureDataHandling:
    """Proper cryptographic practices."""
    
    # BAD: Weak hashing
    def hash_password_insecure(self, password: str) -> str:
        return hashlib.md5(password.encode()).hexdigest()  # NEVER do this!
    
    # GOOD: Secure password hashing
    def hash_password_secure(self, password: str) -> str:
        import argon2
        ph = argon2.PasswordHasher()
        return ph.hash(password)
    
    # BAD: Storing sensitive data in plain text
    def store_credit_card_insecure(self, card_number: str):
        return {'card': card_number}  # Exposed in database!
    
    # GOOD: Encrypt sensitive data
    def store_credit_card_secure(self, card_number: str, key: bytes):
        f = Fernet(key)
        encrypted = f.encrypt(card_number.encode())
        return {'card_encrypted': encrypted.decode()}
    
    # GOOD: Use constant-time comparison for secrets
    def verify_token(self, provided: str, stored: str) -> bool:
        return hmac.compare_digest(provided, stored)

A03: Injection

from sqlalchemy import text
from sqlalchemy.orm import Session
import subprocess
import shlex

class InjectionPrevention:
    """Prevent various injection attacks."""
    
    # BAD: SQL Injection vulnerable
    def get_user_insecure(self, db: Session, username: str):
        query = f"SELECT * FROM users WHERE username = '{username}'"
        return db.execute(query).fetchone()  # Vulnerable!
    
    # GOOD: Parameterized queries
    def get_user_secure(self, db: Session, username: str):
        query = text("SELECT * FROM users WHERE username = :username")
        return db.execute(query, {"username": username}).fetchone()
    
    # BETTER: Use ORM
    def get_user_orm(self, db: Session, username: str):
        return db.query(User).filter(User.username == username).first()
    
    # BAD: Command injection vulnerable
    def ping_host_insecure(self, host: str):
        return subprocess.run(
            f"ping -c 1 {host}",  # Vulnerable to ; rm -rf /
            shell=True,
            capture_output=True
        )
    
    # GOOD: Safe command execution
    def ping_host_secure(self, host: str):
        # Validate input
        import re
        if not re.match(r'^[a-zA-Z0-9.-]+$', host):
            raise ValueError("Invalid hostname")
        
        return subprocess.run(
            ["ping", "-c", "1", host],  # Array form, no shell
            capture_output=True
        )
    
    # BAD: NoSQL Injection vulnerable
    def find_user_nosql_insecure(self, db, query: dict):
        return db.users.find(query)  # {"$gt": ""} bypasses auth!
    
    # GOOD: Validate and sanitize NoSQL queries
    def find_user_nosql_secure(self, db, username: str):
        # Only allow string values, no operators
        if not isinstance(username, str):
            raise ValueError("Invalid input type")
        return db.users.find_one({"username": username})

    # Disable docs in production

# Secure CORS configuration
app.add_middleware(
    CORSMiddleware,
    allow_origins=["https://myapp.com"],  # Not "*"!
    allow_credentials=True,
    allow_methods=["GET", "POST", "PUT", "DELETE"],
    allow_headers=["Authorization", "Content-Type"],
)

# Security headers middleware
@app.middleware("http")
async def add_security_headers(request, call_next):
    response = await call_next(request)
    response.headers["X-Content-Type-Options"] = "nosniff"
    response.headers["X-Frame-Options"] = "DENY"
    response.headers["X-XSS-Protection"] = "1; mode=block"
    response.headers["Content-Security-Policy"] = "default-src 'self'"
    response.headers["Referrer-Policy"] = "strict-origin-when-cross-origin"
    return response

# Don't expose error details in production
@app.exception_handler(Exception)
async def generic_exception_handler(request, exc):
    if PRODUCTION:
        return JSONResponse(
            status_code=500,
            content={"detail": "Internal server error"}
        )
    return JSONResponse(
        status_code=500,
        content={"detail": str(exc)}  # Only in development

A07: Authentication Failures

import jwt
from datetime import datetime, timedelta
import secrets

class SecureAuthentication:
    """Secure authentication implementation."""
    
    def __init__(self, secret_key: str):
        self.secret_key = secret_key
        self.failed_attempts = {}  # In production, use Redis
    
    def authenticate(self, username: str, password: str) -> dict:
        """Authenticate with brute force protection."""
        # Check for account lockout
        lockout_key = f"lockout:{username}"
        if self._is_locked_out(username):
            raise Exception("Account temporarily locked")
        
        user = self._get_user(username)
        
        # Constant-time password verification
        if not user or not self._verify_password(password, user.password_hash):
            self._record_failed_attempt(username)
            # Generic error message (don't reveal if user exists)
            raise Exception("Invalid credentials")
        
        # Clear failed attempts on success
        self._clear_failed_attempts(username)
        
        return self._generate_tokens(user)
    
    def _is_locked_out(self, username: str, max_attempts: int = 5) -> bool:
        attempts = self.failed_attempts.get(username, [])
        recent = [a for a in attempts if a > datetime.now() - timedelta(minutes=15)]
        return len(recent) >= max_attempts
    
    def _generate_tokens(self, user) -> dict:
        """Generate secure JWT tokens."""
        access_token = jwt.encode(
            {
                'sub': str(user.id),
                'type': 'access',
                'exp': datetime.utcnow() + timedelta(minutes=15),
                'jti': secrets.token_hex(16)  # Unique token ID
            },
            self.secret_key,
            algorithm='HS256'
        )
        
        refresh_token = jwt.encode(
            {
                'sub': str(user.id),
                'type': 'refresh',
                'exp': datetime.utcnow() + timedelta(days=7),
                'jti': secrets.token_hex(16)
            },
            self.secret_key,
            algorithm='HS256'
        )
        
        return {
            'access_token': access_token,
            'refresh_token': refresh_token,
            'token_type': 'bearer'
        }

A10: Server-Side Request Forgery (SSRF)

import urllib.parse
import ipaddress
import socket

class SSRFPrevention:
    """Prevent SSRF attacks."""
    
    BLOCKED_HOSTS = {'localhost', '127.0.0.1', '0.0.0.0', '169.254.169.254'}
    BLOCKED_SCHEMES = {'file', 'ftp', 'gopher'}
    
    def validate_url(self, url: str) -> bool:
        """Validate URL to prevent SSRF."""
        parsed = urllib.parse.urlparse(url)
        
        # Check scheme
        if parsed.scheme.lower() in self.BLOCKED_SCHEMES:
            raise ValueError(f"Blocked URL scheme: {parsed.scheme}")
        
        if parsed.scheme.lower() not in ('http', 'https'):
            raise ValueError("Only HTTP/HTTPS allowed")
        
        # Check hostname
        hostname = parsed.hostname
        if not hostname:
            raise ValueError("Invalid URL")
        
        if hostname.lower() in self.BLOCKED_HOSTS:
            raise ValueError("Blocked host")
        
        # Resolve and check IP
        try:
            ip = socket.gethostbyname(hostname)
            ip_obj = ipaddress.ip_address(ip)
            
            if ip_obj.is_private or ip_obj.is_loopback or ip_obj.is_link_local:
                raise ValueError("Private/internal IP addresses blocked")
        except socket.gaierror:
            raise ValueError("Cannot resolve hostname")
        
        return True
    
    def safe_fetch(self, url: str) -> bytes:
        """Safely fetch URL with SSRF protection."""
        self.validate_url(url)
        
        import requests
        response = requests.get(
            url,
            timeout=10,
            allow_redirects=False  # Prevent redirect-based SSRF
        )
        return response.content

Best Practices

  1. Implement defense in depth with multiple security layers
  2. Keep dependencies updated and scan for vulnerabilities regularly
  3. Use security headers for all HTTP responses
  4. Log security events for monitoring and incident response
  5. Validate all input on the server side, never trust client data
  6. Use parameterized queries for all database operations
  7. Implement rate limiting for authentication and sensitive endpoints
  8. Regular security audits and penetration testing
