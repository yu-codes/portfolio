Rate Limiting & Anti-Bot

Protecting APIs with rate limiting and anti-bot measures.

Rate limiting and anti-bot measures protect your APIs from abuse, denial-of-service attacks, and automated threats. These defenses are essential for maintaining service availability and security.

Introduction

Without proper protection, APIs are vulnerable to brute force attacks, credential stuffing, web scraping, and resource exhaustion. Rate limiting controls the rate of requests, while anti-bot measures distinguish legitimate users from automated threats.

Rate Limiting Strategies

AlgorithmDescriptionProsCons
Fixed WindowCount requests per fixed time windowSimple to implementBurst at window edges
Sliding WindowSmoothed request countingNo burst issuesMore memory/computation
Token BucketTokens replenish over timeAllows controlled burstsComplex implementation
Leaky BucketConstant output rateSmooth trafficNo burst allowance

Redis-Based Rate Limiting

import redis
import time
from typing import Tuple
from functools import wraps
from fastapi import HTTPException, Request

class RateLimiter:
    """Distributed rate limiter using Redis."""
    
    def __init__(self, redis_client: redis.Redis):
        self.redis = redis_client
    
    def fixed_window(
        self, 
        key: str, 
        limit: int, 
        window_seconds: int
    ) -> Tuple[bool, int, int]:
        """
        Fixed window rate limiting.
        Returns: (allowed, remaining, reset_time)
        """
        current_window = int(time.time() / window_seconds)
        cache_key = f"rate:{key}:{current_window}"
        
        current_count = self.redis.incr(cache_key)
        if current_count == 1:
            self.redis.expire(cache_key, window_seconds)
        
        remaining = max(0, limit - current_count)
        reset_time = (current_window + 1) * window_seconds
        
        return current_count <= limit, remaining, reset_time
    
    def sliding_window(
        self, 
        key: str, 
        limit: int, 
        window_seconds: int
    ) -> Tuple[bool, int, int]:
        """
        Sliding window rate limiting using sorted sets.
        """
        now = time.time()
        cache_key = f"rate_sliding:{key}"
        window_start = now - window_seconds
        
        pipe = self.redis.pipeline()
        
        # Remove old entries
        pipe.zremrangebyscore(cache_key, 0, window_start)
        
        # Add current request
        pipe.zadd(cache_key, {str(now): now})
        
        # Count requests in window
        pipe.zcard(cache_key)
        
        # Set expiry
        pipe.expire(cache_key, window_seconds)
        
        results = pipe.execute()
        current_count = results[2]
        
        remaining = max(0, limit - current_count)
        reset_time = int(now + window_seconds)
        
        return current_count <= limit, remaining, reset_time
    
    def token_bucket(
        self, 
        key: str, 
        capacity: int, 
        refill_rate: float
    ) -> Tuple[bool, int]:
        """
        Token bucket algorithm.
        capacity: max tokens
        refill_rate: tokens per second
        """
        cache_key = f"bucket:{key}"
        now = time.time()
        
        # Get current bucket state
        bucket_data = self.redis.hgetall(cache_key)
        
        if bucket_data:
            tokens = float(bucket_data.get(b'tokens', capacity))
            last_update = float(bucket_data.get(b'last_update', now))
            
            # Calculate tokens to add
            time_passed = now - last_update
            tokens = min(capacity, tokens + time_passed * refill_rate)
        else:
            tokens = capacity
        
        # Try to consume a token
        if tokens >= 1:
            tokens -= 1
            allowed = True
        else:
            allowed = False
        
        # Update bucket state
        self.redis.hset(cache_key, mapping={
            'tokens': tokens,
            'last_update': now
        })
        self.redis.expire(cache_key, int(capacity / refill_rate) + 1)
        
        return allowed, int(tokens)


def rate_limit(
    limit: int = 100, 
    window: int = 60,
    key_func: callable = None
):
    """Rate limiting decorator for FastAPI."""
    def decorator(func):
        @wraps(func)
        async def wrapper(request: Request, *args, **kwargs):
            # Determine rate limit key
            if key_func:
                key = key_func(request)
            else:
                key = request.client.host
            
            limiter = request.app.state.rate_limiter
            allowed, remaining, reset = limiter.sliding_window(
                key, limit, window
            )
            
            # Add rate limit headers
            headers = {
                "X-RateLimit-Limit": str(limit),
                "X-RateLimit-Remaining": str(remaining),
                "X-RateLimit-Reset": str(reset)
            }
            
            if not allowed:
                raise HTTPException(
                    status_code=429,
                    detail="Rate limit exceeded",
                    headers=headers
                )
            
            response = await func(request, *args, **kwargs)
            
            # Add headers to response
            for header, value in headers.items():
                response.headers[header] = value
            
            return response
        return wrapper
    return decorator

NGINX Rate Limiting

http {
    # Define rate limit zones
    limit_req_zone $binary_remote_addr zone=api_limit:10m rate=10r/s;
    limit_req_zone $binary_remote_addr zone=login_limit:10m rate=1r/s;
    limit_req_zone $http_authorization zone=user_limit:10m rate=100r/s;
    
    # Connection limiting
    limit_conn_zone $binary_remote_addr zone=conn_limit:10m;
    
    server {
        listen 443 ssl;
        server_name api.example.com;
        
        # Global connection limit
        limit_conn conn_limit 100;
        
        # API endpoints
        location /api/ {
            limit_req zone=api_limit burst=20 nodelay;
            limit_req_status 429;
            
            proxy_pass http://backend;
        }
        
        # Login endpoint - stricter limits
        location /api/login {
            limit_req zone=login_limit burst=5;
            limit_req_status 429;
            
            proxy_pass http://backend;
        }
        
        # Rate limit response
        error_page 429 = @rate_limit_exceeded;
        location @rate_limit_exceeded {
            default_type application/json;
            return 429 '{"error": "Rate limit exceeded", "retry_after": 60}';
        }
    }
}

Bot Detection

from fastapi import Request, HTTPException
import hashlib
import re
from typing import List, Dict
import ipaddress

class BotDetector:
    """Detect and block bot traffic."""
    
    def __init__(self):
        # Known bot user agents (partial matches)
        self.bot_patterns = [
            r'bot', r'crawler', r'spider', r'scraper',
            r'curl', r'wget', r'python-requests',
            r'headless', r'phantom', r'selenium'
        ]
        
        # Known bad IPs/ranges (example)
        self.blocked_ranges = [
            # Add known malicious IP ranges
        ]
        
        # Request patterns indicating bots
        self.suspicious_patterns = {
            'missing_headers': ['Accept', 'Accept-Language'],
            'suspicious_header_values': {
                'Accept-Encoding': ['identity'],
            }
        }
    
    def check_user_agent(self, user_agent: str) -> bool:
        """Check if user agent matches known bot patterns."""
        if not user_agent:
            return True  # Missing UA is suspicious
        
        ua_lower = user_agent.lower()
        for pattern in self.bot_patterns:
            if re.search(pattern, ua_lower):
                return True
        return False
    
    def check_ip(self, ip: str) -> bool:
        """Check if IP is in blocked ranges."""
        try:
            ip_obj = ipaddress.ip_address(ip)
            for blocked_range in self.blocked_ranges:
                if ip_obj in ipaddress.ip_network(blocked_range):
                    return True
        except ValueError:
            return True  # Invalid IP is suspicious
        return False
    
    def check_request_patterns(self, request: Request) -> float:
        """
        Score request based on suspicious patterns.
        Returns score 0-1, higher = more suspicious.
        """
        score = 0.0
        
        # Check for missing common headers
        for header in self.suspicious_patterns['missing_headers']:
            if header not in request.headers:
                score += 0.2
        
        # Check header values
        for header, suspicious_values in self.suspicious_patterns['suspicious_header_values'].items():
            value = request.headers.get(header, '')
            if value in suspicious_values:
                score += 0.3
        
        # Check for consistent header ordering (bots often have non-standard order)
        # Real browsers have consistent header ordering
        
        return min(1.0, score)
    
    def analyze_request(self, request: Request) -> Dict:
        """Comprehensive request analysis."""
        user_agent = request.headers.get('User-Agent', '')
        client_ip = request.client.host
        
        analysis = {
            'is_bot_ua': self.check_user_agent(user_agent),
            'is_blocked_ip': self.check_ip(client_ip),
            'suspicion_score': self.check_request_patterns(request),
            'client_ip': client_ip,
            'user_agent': user_agent
        }
        
        analysis['is_likely_bot'] = (
            analysis['is_bot_ua'] or 
            analysis['is_blocked_ip'] or 
            analysis['suspicion_score'] > 0.5
        )
        
        return analysis


class CAPTCHAChallenge:
    """CAPTCHA challenge for suspicious requests."""
    
    def __init__(self, recaptcha_secret: str):
        self.secret = recaptcha_secret
        self.verify_url = "https://www.google.com/recaptcha/api/siteverify"
    
    async def verify(self, token: str, remote_ip: str) -> bool:
        """Verify reCAPTCHA response."""
        import aiohttp
        
        async with aiohttp.ClientSession() as session:
            async with session.post(
                self.verify_url,
                data={
                    'secret': self.secret,
                    'response': token,
                    'remoteip': remote_ip
                }
            ) as response:
                result = await response.json()
                return result.get('success', False)


# Middleware for bot detection
@app.middleware("http")
async def bot_detection_middleware(request: Request, call_next):
    bot_detector = request.app.state.bot_detector
    analysis = bot_detector.analyze_request(request)
    
    if analysis['is_blocked_ip']:
        raise HTTPException(status_code=403, detail="Access denied")
    
    if analysis['is_likely_bot'] and not request.url.path.startswith('/api/public'):
        # Challenge suspicious requests
        captcha_token = request.headers.get('X-Captcha-Token')
        if not captcha_token:
            raise HTTPException(
                status_code=403,
                detail="CAPTCHA verification required",
                headers={"X-Captcha-Required": "true"}
            )
    
    return await call_next(request)

Fingerprinting

import hashlib
from typing import Optional

class DeviceFingerprint:
    """Generate device fingerprint for tracking."""
    
    def generate(self, request: Request) -> str:
        """Generate fingerprint from request headers."""
        components = [
            request.headers.get('User-Agent', ''),
            request.headers.get('Accept-Language', ''),
            request.headers.get('Accept-Encoding', ''),
            request.headers.get('Accept', ''),
            # Add more headers as needed
        ]
        
        fingerprint_string = '|'.join(components)
        return hashlib.sha256(fingerprint_string.encode()).hexdigest()[:16]
    
    def track_fingerprint(
        self, 
        fingerprint: str, 
        ip: str,
        redis_client
    ) -> dict:
        """Track fingerprint behavior over time."""
        key = f"fingerprint:{fingerprint}"
        
        # Get or initialize tracking data
        data = redis_client.hgetall(key) or {}
        
        # Update tracking
        current_count = int(data.get(b'request_count', 0)) + 1
        ips_seen = set(data.get(b'ips', b'').decode().split(','))
        ips_seen.add(ip)
        
        redis_client.hset(key, mapping={
            'request_count': current_count,
            'ips': ','.join(filter(None, ips_seen)),
            'last_seen': time.time()
        })
        redis_client.expire(key, 86400)  # 24 hours
        
        return {
            'fingerprint': fingerprint,
            'request_count': current_count,
            'unique_ips': len(ips_seen),
            'is_suspicious': current_count > 1000 or len(ips_seen) > 10
        }

Rate Limit Bypass Prevention

AttackDescriptionPrevention
IP RotationUsing multiple IPsRate limit by fingerprint + API key
Header SpoofingFake X-Forwarded-ForTrust only first proxy hop
Account CreationFresh accounts for each requestRate limit account creation
Distributed AttackMany IPs, low rate eachGlobal rate limits, anomaly detection

Best Practices

  1. Layer your defenses with multiple rate limiting strategies
  2. Use sliding windows over fixed windows to prevent burst abuse
  3. Rate limit by multiple keys (IP, user ID, API key, fingerprint)
  4. Return proper headers (X-RateLimit-*) for client awareness
  5. Implement gradual responses before hard blocking
  6. Monitor and adjust limits based on traffic patterns
  7. Whitelist legitimate bots (Googlebot, etc.) by verified IP
  8. Log blocked requests for security analysis
